#alias 153: TYPE_TUBEPATH




sub ObjectMain
	switch Object.PropertyValue
	case 0
		foreach TypeGroup[256],PlayerObjectPos
			if Object[PlayerObjectPos].State!=1
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if CheckResult==1
					Object[PlayerObjectPos].XPos=Object.XPos
					Object[PlayerObjectPos].State=1
					Object[PlayerObjectPos].TileCollisions=0
					if Object[PlayerObjectPos].Animation!=ANI_JUMPING
						PlaySfx(SFXName[Rolling],0)
						Object[PlayerObjectPos].Animation=ANI_JUMPING
					endif
					CreateTempObject(TypeName[TubePath],2,Object.XPos,Object.YPos)
					Object[TempObjectPos].Value1=PlayerObjectPos
					Object[TempObjectPos].Value2=Object.EntityNo
					Object[TempObjectPos].Value2++
				endif
			endif
		floop
		break
	case 1
		foreach TypeGroup[256],PlayerObjectPos
			if Object[PlayerObjectPos].State==1
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0,0,0,0)
				if CheckResult==1
					Object[PlayerObjectPos].State=12
					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
					Object[PlayerObjectPos].XPos=Object.XPos
					Object[PlayerObjectPos].YPos=Object.YPos
					Object[PlayerObjectPos].YVelocity=-0xA0000
					Object[PlayerObjectPos].Speed=0
					Object[PlayerObjectPos].XVelocity=0
					Object[PlayerObjectPos].TileCollisions=1
					PlaySfx(SFXName[Release],0)
				endif
			endif
		floop
		break
	case 2
		PlayerObjectPos=Object.Value1
		if Object[PlayerObjectPos].State==1
			ArrayPos0=Object.Value2
			TempValue0=Object[PlayerObjectPos].XPos
			TempValue0-=Object[ArrayPos0].XPos
			TempValue0>>=16
			TempValue1=Object[PlayerObjectPos].YPos
			TempValue1-=Object[ArrayPos0].YPos
			TempValue1>>=16
			ATan2(TempValue2,TempValue0,TempValue1)
			Cos256(Object[PlayerObjectPos].XVelocity,TempValue2)
			Sin256(Object[PlayerObjectPos].YVelocity,TempValue2)
			Object[PlayerObjectPos].XVelocity*=-0xC00
			Object[PlayerObjectPos].YVelocity*=-0xC00
			TempValue2=Object[PlayerObjectPos].XVelocity
			Absolute(TempValue2)
			TempValue3=Object[PlayerObjectPos].YVelocity
			Absolute(TempValue3)
			if TempValue2>TempValue3
				Object[PlayerObjectPos].YVelocity=0
			else
				Object[PlayerObjectPos].XVelocity=0
			endif
			if Object[ArrayPos0].Type==TypeName[BlankObject]
				TempValue2=TempValue0
				TempValue2*=TempValue0
				TempValue0=TempValue1
				TempValue0*=TempValue1
				TempValue0+=TempValue2
				if TempValue0<256
					Object[PlayerObjectPos].XPos=Object[ArrayPos0].XPos
					Object[PlayerObjectPos].YPos=Object[ArrayPos0].YPos
					Object[PlayerObjectPos].XVelocity=0
					Object[PlayerObjectPos].YVelocity=0
					Object.Value2++
				endif
			endif
		else
			Object.Type=TypeName[BlankObject]
		endif
		break

	endswitch
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
