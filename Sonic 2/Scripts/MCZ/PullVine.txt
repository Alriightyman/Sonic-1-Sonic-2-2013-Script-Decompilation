#function PullVine_Function124
#function PullVine_Function125



function PullVine_Function124
	DrawSprite(1)
endfunction


function PullVine_Function125
	CreateTempObject(TypeName[PullVine],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value1=Object[TempObjectPos].YPos
	Object[TempObjectPos].YVelocity=0x20000
	Object[TempObjectPos].Value2=Object[TempObjectPos].YPos
endfunction


sub ObjectMain
	if Object.PropertyValue==1
		if Object.Value0<88
			Object.Value0++
			Object.YPos+=Object.YVelocity
		endif
	else
		if Object.Value0>0
			Object.Value0--
			Object.YPos-=Object.YVelocity
		endif
	endif
	Object.PropertyValue=0
	if Object.Value3>0
		Object.Value3--
	endif
	TempValue0=Object.YPos
	TempValue0+=0x380000
	foreach TypeGroup[256],PlayerObjectPos
		if Object[PlayerObjectPos].State!=1
			if Object[PlayerObjectPos].Down==0
				if Object.Value3==0
					if Object[PlayerObjectPos].YPos<TempValue0
						PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,88,8,96,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
					else
						PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,40,8,48,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
					endif
					if CheckResult==1
						if Object[PlayerObjectPos].Gravity==GRAVITY_AIR
							Object[PlayerObjectPos].State=1
							Object[PlayerObjectPos].Animation=ANI_HANGING
							Object[PlayerObjectPos].Speed=0
							Object[PlayerObjectPos].XVelocity=0
							Object[PlayerObjectPos].YVelocity=0
							Object[PlayerObjectPos].XPos=Object.XPos
							Object[PlayerObjectPos].YPos=Object[PlayerObjectPos].CollisionTop
							FlipSign(Object[PlayerObjectPos].YPos)
							Object[PlayerObjectPos].YPos+=36
							Object[PlayerObjectPos].YPos<<=16
							Object[PlayerObjectPos].YPos+=Object.YPos
							Object.PropertyValue|=1
						endif
					endif
				endif
			endif
		else
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-4,8,72,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				if Object[PlayerObjectPos].JumpPress==1
					Object[PlayerObjectPos].State=12
					Object[PlayerObjectPos].YVelocity=-0x40000
					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
					Object.Value3=30
				else
					Object[PlayerObjectPos].XPos=Object.XPos
					Object[PlayerObjectPos].YPos=Object[PlayerObjectPos].CollisionTop
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos+=36
					Object[PlayerObjectPos].YPos<<=16
					Object[PlayerObjectPos].YPos+=Object.YPos
					Object.PropertyValue|=1
				endif
			endif
		endif
	floop
	if Object.Frame==0
		if Object.PropertyValue==1
			PlaySfx(SFXName[Catch],0)
		endif
	endif
	Object.Frame=Object.PropertyValue
endsub


sub ObjectDraw
	TempValue0=Object.YPos
	TempValue0-=Object.Value2
	TempValue0>>=16
	TempValue1=176
	TempValue1-=TempValue0
	TempValue2=-40
	TempValue2-=TempValue0
	TempValue0+=80
	EditFrame(0,-12,TempValue2,24,TempValue0,232,TempValue1)
	DrawSprite(0)
endsub


sub ObjectStartup
	LoadSpriteSheet("MCZ/Objects.gif")
	SpriteFrame(-12,-40,24,80,232,176)
	SpriteFrame(-12,-40,24,80,232,176)
	foreach TypeName[PullVine],ArrayPos0
		Object[ArrayPos0].Value2=Object[ArrayPos0].YPos
		if Object[ArrayPos0].PropertyValue==0
			Object[ArrayPos0].YVelocity=0x20000
		else
			Object[ArrayPos0].YVelocity=-0x20000
			Object[ArrayPos0].YPos+=0xB00000
		endif
		Object[ArrayPos0].Value1=Object[ArrayPos0].YPos
		Object[ArrayPos0].PropertyValue=0
	floop
	SetArrayValue(51,DebugMode_constant25,DebugMode_array12)
	SetArrayValue(124,DebugMode_constant25,DebugMode_array13)
	SetArrayValue(125,DebugMode_constant25,DebugMode_array11)
	DebugMode_constant25++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
