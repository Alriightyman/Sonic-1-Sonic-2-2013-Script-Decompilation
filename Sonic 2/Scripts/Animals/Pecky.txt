function Pecky_Function106
	CreateTempObject(TypeName[Pecky],0,Object.XPos,Object.YPos)
	GetBit(TempValue0,Object.Direction,0)
	if TempValue0==0
		Object[TempObjectPos].State=3
		Object[TempObjectPos].XVelocity=0x18000
		Object[TempObjectPos].YVelocity=-0x30000
		Rand(CheckResult,32)
		if CheckResult<16
			Object[TempObjectPos].Direction=FLIP_X
			FlipSign(Object[TempObjectPos].XVelocity)
		endif
	else
		Object[TempObjectPos].State=4
	endif

sub ObjectMain
	switch Object.State
	case 0
		if Object.Value0<Object.PropertyValue
			Object.Value0++
		else
			Object.XVelocity=0x18000
			if Object.Value1==1
				Object.DrawOrder=4
				Rand(CheckResult,32)
				if CheckResult<16
					Object.Direction=FLIP_X
					FlipSign(Object.XVelocity)
				endif
			else
				Object.Direction=FLIP_X
				FlipSign(Object.XVelocity)
			endif
			Object.YVelocity=-0x40000
			Object.State++
		endif
		break
	case 1
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,12,0)
			if CheckResult==1
				Object.YVelocity=-0x30000
				Object.State++
			endif
		endif
		break
	case 4
		if Object[0].XPos<Object.XPos
			Object.Direction=FLIP_X
		else
			Object.Direction=FLIP_NONE
		endif
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,12,0)
			if CheckResult==1
				if Object.State==3
					if Object.Value0==1
						FlipSign(Object.XVelocity)
						Object.Direction^=FLIP_X
					endif
					Object.Value0^=1
				endif
				Object.YVelocity=-0x30000
			endif
		endif
		break
	case 2
	case 3
	case 5
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,12,0)
			if CheckResult==1
				if Object.State==3
					if Object.Value0==1
						FlipSign(Object.XVelocity)
						Object.Direction^=FLIP_X
					endif
					Object.Value0^=1
				endif
				Object.YVelocity=-0x30000
			endif
		endif
		break

	endswitch
	if Object.State<3
		if Object.OutOfBounds==1
			Object.Type=TypeName[BlankObject]
		endif
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 0
	case 1
		DrawSprite(2)
		break
	case 2
	case 3
	case 4
	case 5
		if Object.YVelocity<0
			DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		else
			DrawSpriteFX(1,FX_FLIP,Object.XPos,Object.YPos)
		endif
		break

	endswitch
endsub
