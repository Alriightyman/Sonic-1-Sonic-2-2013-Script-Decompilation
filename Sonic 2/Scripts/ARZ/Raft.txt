function Raft_Function123
	CreateTempObject(TypeName[Raft],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Direction=Object.Direction
endfunction


sub ObjectMain
	TempValue0=Object.XPos
	TempValue0&=-0x10000
	TempValue1=Object.YPos
	TempValue1&=-0x10000
	TempValue7=Stage.WaterLevel
	TempValue7<<=16
	if Object.State==1
		Object.YPos=TempValue7
	else
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x1800
		if Object.YPos>=TempValue7
			if Object.Direction==FLIP_NONE
				if Object.XVelocity>0x10000
					Object.XVelocity=0x10000
				endif
			else
				if Object.XVelocity<-0x10000
					Object.XVelocity=-0x10000
				endif
			endif
			Object.State++
		endif
	endif
	Object.XPos+=Object.XVelocity
	if Object.Direction==FLIP_NONE
		ObjectTileCollision(CMODE_LWALL,32,12,0)
	else
		ObjectTileCollision(CMODE_RWALL,-32,12,0)
	endif
	TempValue4=Object.XPos
	TempValue5=Object.YPos
	TempValue2=Object.XPos
	TempValue2&=-0x10000
	TempValue2-=TempValue0
	TempValue3=Object.YPos
	TempValue3&=-0x10000
	TempValue3-=TempValue1
	Object.XPos=TempValue0
	Object.YPos=TempValue1
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-32,-8,32,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		if CheckResult==1
			if Object.XVelocity==0
				if Object.Direction==FLIP_NONE
					Object.XVelocity=0x10000
				else
					Object.XVelocity=-0x10000
				endif
			endif
			Object[PlayerObjectPos].XPos+=TempValue2
			Object[PlayerObjectPos].YPos+=TempValue3
		endif
	floop
	Object.XPos=TempValue4
	Object.YPos=TempValue5
	if Object.OutOfBounds==1
		if Object.Value0==0
			Object.Type=TypeName[BlankObject]
		else
			Object.XPos=Object.Value0
			Object.YPos=Object.Value1
			if Object.OutOfBounds==0
				Object.XPos=TempValue4
				Object.YPos=TempValue5
			endif
		endif
	endif
endsub


sub ObjectDraw
	DrawSprite(0)
endsub

	SetArrayValue(51,DebugMode_constant25,DebugMode_array12)
	SetArrayValue(122,DebugMode_constant25,DebugMode_array13)
	SetArrayValue(123,DebugMode_constant25,DebugMode_array11)
	DebugMode_constant25++
	foreach TypeName[Raft],ArrayPos0
		Object[ArrayPos0].Value0=Object[ArrayPos0].XPos
		Object[ArrayPos0].Value1=Object[ArrayPos0].YPos
	floop
endsub
