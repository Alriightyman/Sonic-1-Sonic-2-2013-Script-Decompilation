function ChemicalDrop_Function165
	if Object[PlayerObjectPos].Value37!=2
		if Object[PlayerObjectPos].State!=28
			ArrayPos0=Object[PlayerObjectPos].EntityNo
			ArrayPos0+=PlayerObjectCount
			if Object[PlayerObjectPos].Value7==0
				if Object[PlayerObjectPos].Value8==0
					Object[PlayerObjectPos].State=26
					if Object[PlayerObjectPos].XPos>Object.XPos
						Object[PlayerObjectPos].Speed=0x20000
					else
						Object[PlayerObjectPos].Speed=-0x20000
					endif
				endif
			endif
		endif
	endif
endfunction


function ChemicalDrop_Function166
	TempValue1=Object.YVelocity
	TempValue1>>=1
	FlipSign(TempValue1)
	PlaySfx(SFXName[ChemicalSplash],0)
	Object.Type=TypeName[BlankObject]
	TempValue0=0
	while TempValue0<5
		CreateTempObject(TypeName[ChemicalSplash],0,Object.XPos,Object.YPos)
		Object[TempObjectPos].DrawOrder=2
		Rand(Object[TempObjectPos].XVelocity,512)
		Object[TempObjectPos].XVelocity-=256
		Object[TempObjectPos].XVelocity<<=10
		if Object[TempObjectPos].XVelocity<0
			Object[TempObjectPos].XVelocity+=0x20000
		endif
		Object[TempObjectPos].XVelocity-=0x10000
		if TempValue0>0
			Rand(Object[TempObjectPos].YVelocity,0x7FE)
			Object[TempObjectPos].YVelocity-=0x400
			Object[TempObjectPos].YVelocity<<=8
		endif
		Object[TempObjectPos].YVelocity+=TempValue1
		TempValue0++
	loop
endfunction


sub ObjectMain
	Object.YPos+=Object.YVelocity
	Object.YVelocity+=0x3800
	TempValue1=Object.YPos
	TempValue1>>=16
	TempValue1-=Screen.CameraYPos
	Absolute(TempValue1)
	TempValue1-=15
	if TempValue1>=Screen.CenterY
		Object.Type=TypeName[BlankObject]
	endif
	ObjectTileCollision(CMODE_FLOOR,0,11,0)
	if CheckResult==1
		CallFunction(ChemicalDrop_Function166)
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-6,-6,6,6,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		if CheckResult==1
			CallFunction(ChemicalDrop_Function165)
			CallFunction(ChemicalDrop_Function166)
		endif
	floop
endsub
