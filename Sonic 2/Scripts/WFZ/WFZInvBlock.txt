#alias 87: TYPE_WFZINVBLOCK




sub ObjectMain
	TempValue0=Object.Value0
	FlipSign(TempValue0)
	TempValue1=Object.Value1
	FlipSign(TempValue1)
	foreach TypeGroup[256],PlayerObjectPos
		if Object[PlayerObjectPos].State!=25
			if Object[PlayerObjectPos].Animation!=ANI_FANROTATE
				PlayerObjectCollision(C_BOX,Object.EntityNo,TempValue0,TempValue1,Object.Value0,Object.Value1,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				switch CheckResult
				case 0
					TempValue0+=2
					TempValue1+=2
					TempValue2=Object.Value0
					TempValue3=Object.Value1
					TempValue2-=2
					TempValue3-=2
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,TempValue0,TempValue1,TempValue2,TempValue3,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
					if CheckResult==1
						Object[PlayerObjectPos].Gravity=GRAVITY_GROUND
					endif
					break
				case 4
					if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
						CallFunction(PlayerObject_Function50)
					endif
					break

				endswitch
			endif
		endif
	floop
endsub


sub ObjectDraw
	if Object[0].Type==TypeName[DebugMode]
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		TempValue2=Object.PropertyValue
		TempValue4=TempValue2
		TempValue4&=240
		TempValue4<<=15
		TempValue2>>=4
		TempValue2++
		TempValue6=TempValue2
		TempValue3=Object.PropertyValue
		TempValue3&=15
		TempValue5=TempValue3
		TempValue5<<=19
		TempValue3++
		TempValue0-=TempValue4
		TempValue4=TempValue0
		TempValue1-=TempValue5
		while TempValue3>0
			while TempValue2>0
				DrawSpriteXY(0,TempValue0,TempValue1)
				TempValue0+=0x100000
				TempValue2--
			loop
			TempValue2=TempValue6
			TempValue0=TempValue4
			TempValue1+=0x100000
			TempValue3--
		loop
	endif
endsub


sub ObjectStartup
	LoadSpriteSheet("Global/Display.gif")
	SpriteFrame(-8,-7,16,14,1,176)
	foreach TypeName[WFZInvBlock],ArrayPos0
		if Object[ArrayPos0].Value0==1
			if options.gameMode!=2
				Object[ArrayPos0].Type=TypeName[BlankObject]
			endif
		endif
		Object[ArrayPos0].DrawOrder=5
		Object[ArrayPos0].Value0=Object[ArrayPos0].PropertyValue
		Object[ArrayPos0].Value0&=240
		Object[ArrayPos0].Value0+=16
		Object[ArrayPos0].Value0>>=1
		Object[ArrayPos0].Value1=Object[ArrayPos0].PropertyValue
		Object[ArrayPos0].Value1&=15
		Object[ArrayPos0].Value1++
		Object[ArrayPos0].Value1<<=3
	floop
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
