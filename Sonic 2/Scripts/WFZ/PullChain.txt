#function PullChain_Function119
#function PullChain_Function120

#constant 0:PullChain_constant45


function PullChain_Function119
	DrawSprite(1)
endfunction


function PullChain_Function120
	TempValue0=Object[0].Value17
	TempValue0-=PullChain_constant44
	CreateTempObject(TypeName[PullChain],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value2=Object[TempObjectPos].YPos
	GetBit(TempValue1,TempValue0,0)
	if TempValue1==0
		Object[TempObjectPos].Value4=80
	else
		Object[TempObjectPos].Value4=48
	endif
	GetBit(TempValue2,TempValue0,1)
	if TempValue2==0
		Object[TempObjectPos].YVelocity=0x20000
		Object[TempObjectPos].YPos-=0xF0000
	else
		Object[TempObjectPos].YVelocity=-0x20000
		if TempValue1==0
			Object[TempObjectPos].YPos+=0x910000
		else
			Object[TempObjectPos].YPos+=0x510000
		endif
	endif
	Object[TempObjectPos].Value1=Object[TempObjectPos].YPos
endfunction


sub ObjectMain
	if Object.PropertyValue==1
		if Object.Value0<Object.Value4
			Object.Value0++
			Object.YPos+=Object.YVelocity
		endif
	else
		if Object.Value0>0
			Object.Value0--
			Object.YPos-=Object.YVelocity
		endif
	endif
	Object.PropertyValue=0
	if Object.Value3>0
		Object.Value3--
	endif
	TempValue0=Object.YPos
	TempValue0+=0x380000
	foreach TypeGroup[256],PlayerObjectPos
		if Object[PlayerObjectPos].State!=1
			if Object.Value3==0
				if Object[PlayerObjectPos].YPos<TempValue0
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,70,8,96,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				else
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,32,8,48,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				endif
				if CheckResult==1
					Object[PlayerObjectPos].State=1
					Object[PlayerObjectPos].Animation=ANI_HANGING
					Object[PlayerObjectPos].Speed=0
					Object[PlayerObjectPos].XVelocity=0
					Object[PlayerObjectPos].YVelocity=0
					Object[PlayerObjectPos].XPos=Object.XPos
					Object[PlayerObjectPos].YPos=Object[PlayerObjectPos].CollisionTop
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos+=36
					Object[PlayerObjectPos].YPos<<=16
					Object[PlayerObjectPos].YPos+=Object.YPos
					Object.PropertyValue|=1
				endif
			endif
		else
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-4,8,72,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				if Object[PlayerObjectPos].JumpPress==1
					Object[PlayerObjectPos].State=12
					Object[PlayerObjectPos].YVelocity=-0x40000
					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
					Object.Value3=30
				else
					Object[PlayerObjectPos].XPos=Object.XPos
					Object[PlayerObjectPos].YPos=Object[PlayerObjectPos].CollisionTop
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos+=36
					Object[PlayerObjectPos].YPos<<=16
					Object[PlayerObjectPos].YPos+=Object.YPos
					Object.PropertyValue|=1
				endif
			endif
		endif
	floop
	if Object.Frame==0
		if Object.PropertyValue==1
			PlaySfx(SFXName[Catch],0)
		endif
	endif
	Object.Frame=Object.PropertyValue
endsub


sub ObjectDraw
	TempValue0=Object.YPos
	TempValue0-=Object.Value2
	TempValue0>>=16
	TempValue1=PullChain_constant45
	TempValue1-=TempValue0
	TempValue2=-40
	TempValue2-=TempValue0
	TempValue0+=80
	EditFrame(0,-12,TempValue2,24,TempValue0,PullChain_constant46,TempValue1)
	DrawSprite(0)
endsub


sub ObjectStartup
	CheckStageFolder("Zone11")
	if CheckResult==1
		LoadSpriteSheet("SCZ/Objects.gif")
		SpriteFrame(-12,-40,24,65,236,191)
		SpriteFrame(-12,-40,24,65,236,191)
		PullChain_constant46=236
		PullChain_constant45=176
	else
		LoadSpriteSheet("MBZ/Objects.gif")
		SpriteFrame(-12,-40,24,65,901,358)
		SpriteFrame(-12,-40,24,65,901,358)
		PullChain_constant46=901
		PullChain_constant45=358
		PullChain_constant45+=176
	endif
	foreach TypeName[PullChain],ArrayPos0
		Object[ArrayPos0].Value2=Object[ArrayPos0].YPos
		GetBit(TempValue1,Object[ArrayPos0].PropertyValue,0)
		if TempValue1==0
			Object[ArrayPos0].Value4=80
		else
			Object[ArrayPos0].Value4=48
		endif
		GetBit(TempValue0,Object[ArrayPos0].PropertyValue,4)
		if TempValue0==0
			Object[ArrayPos0].YVelocity=0x20000
			Object[ArrayPos0].YPos-=0x200000
		else
			Object[ArrayPos0].YVelocity=-0x20000
			if TempValue1==0
				Object[ArrayPos0].YPos+=0x800000
			else
				Object[ArrayPos0].YPos+=0x400000
			endif
			Object[ArrayPos0].Value2-=0x110000
		endif
		Object[ArrayPos0].Value1=Object[ArrayPos0].YPos
		Object[ArrayPos0].PropertyValue=0
	floop
	TempValue0=0
	PullChain_constant44=DebugMode_constant25
	while TempValue0<4
		SetArrayValue(54,DebugMode_constant25,DebugMode_array12)
		SetArrayValue(119,DebugMode_constant25,DebugMode_array13)
		SetArrayValue(120,DebugMode_constant25,DebugMode_array11)
		DebugMode_constant25++
		TempValue0++
	loop
endsub
