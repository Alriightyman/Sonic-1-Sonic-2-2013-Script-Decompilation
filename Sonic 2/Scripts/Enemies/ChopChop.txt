function ChopChop_Function148
	Object[ArrayPos0].Direction=Object[ArrayPos0].PropertyValue
	if Object[ArrayPos0].Direction==FLIP_NONE
		Object[ArrayPos0].XVelocity=-0x4000
	else
		Object[ArrayPos0].XVelocity=0x4000
	endif
	Object[ArrayPos0].YVelocity=0
	Object[ArrayPos0].Value0=512
	Object[ArrayPos0].Value1=0
	Object[ArrayPos0].Frame=0
	Object[ArrayPos0].Animation=0
	Object[ArrayPos0].AnimationTimer=0
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction


function ChopChop_Function149
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function ChopChop_Function150
	CreateTempObject(TypeName[ChopChop],Object.Direction,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value2=Object.XPos
	Object[TempObjectPos].Value3=Object.YPos
	ArrayPos0=Object[TempObjectPos].EntityNo
	CallFunction(ChopChop_Function148)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.XPos+=Object.XVelocity
		Object.Value0--
		if Object.Value0==0
			Object.Direction^=FLIP_X
			FlipSign(Object.XVelocity)
			Object.Value0=512
		endif
		foreach TypeGroup[256],PlayerObjectPos
			if Object.State==0
				if Object.Direction==FLIP_NONE
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-160,-32,16,32,PlayerObjectPos,-1,-1,1,1)
				else
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,16,-32,160,32,PlayerObjectPos,-1,-1,1,1)
				endif
				if CheckResult==1
					Object.State=1
					Object.Animation=1
					Object.Value0=16
					if Object[PlayerObjectPos].XPos<Object.XPos
						Object.XVelocity=-0x20000
					else
						Object.XVelocity=0x20000
					endif
					TempValue0=Object.YPos
					TempValue0-=Object[PlayerObjectPos].YPos
					Absolute(TempValue0)
					if TempValue0>0x100000
						if Object[PlayerObjectPos].YPos<Object.YPos
							Object.YVelocity=-0x8000
						else
							Object.YVelocity=0x8000
						endif
					endif
				endif
			endif
		floop
		Object.Value1++
		if Object.Value1>=240
			Object.Value1=0
		endif
		break
	case 1
		Object.Value0--
		if Object.Value0<0
			Object.State=2
		endif
		break
	case 2
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.Value1++
		if Object.Value1>=20
			Object.Value1=0
		endif
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value2
		Object.YPos=Object.Value3
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(ChopChop_Function148)
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	if Object.Animation==1
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=3
		Object.AnimationTimer++
		Object.AnimationTimer&=15
	endif
	if Object.Value1==0
		CreateTempObject(TypeName[AirBubble],2,Object.XPos,Object.YPos)
		Object[TempObjectPos].DrawOrder=4
		Object[TempObjectPos].YVelocity=-0x8800
		Object[TempObjectPos].YPos+=0x60000
		if Object.Direction==FLIP_NONE
			Object[TempObjectPos].XPos+=0x100000
		else
			Object[TempObjectPos].XPos-=0x100000
			Object[TempObjectPos].Angle=256
		endif
		Object[TempObjectPos].Value2=0xFFFF
		Object[TempObjectPos].Value1=Object[TempObjectPos].XPos
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-12,16,12,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub


sub ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
endsub
