function Whisp_Function145
	Object[ArrayPos0].XVelocity=0
	Object[ArrayPos0].YVelocity=0
	Object[ArrayPos0].Value0=16
	Object[ArrayPos0].Value1=4
	Object[ArrayPos0].AnimationTimer=0
	Object[ArrayPos0].InkEffect=INK_ALPHA
	Object[ArrayPos0].Alpha=192
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
	Object.State=0
endfunction


function Whisp_Function146
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Whisp_Function147
	CreateTempObject(TypeName[Whisp],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value2=Object.XPos
	Object[TempObjectPos].Value3=Object.YPos
	ArrayPos0=Object[TempObjectPos].EntityNo
	CallFunction(Whisp_Function145)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.Value0--
		if Object.Value0<0
			Object.Value1--
			if Object.Value1>=0
				Object.YVelocity=-0x10000
				Object.Value0=96
				Object.State=1
			else
				Object.XVelocity=-0x20000
				Object.YVelocity=-0x20000
				Object.State=2
			endif
		endif
		break
	case 1
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		TempValue2=0x7FFFFFFF
		ArrayPos0=0
		foreach TypeGroup[256],PlayerObjectPos
			TempValue0=Object[PlayerObjectPos].XPos
			TempValue0-=Object.XPos
			if TempValue0<0
				TempValue1=0
				FlipSign(TempValue0)
			else
				TempValue1=1
			endif
			if TempValue0<TempValue2
				TempValue2=TempValue0
				ArrayPos0=PlayerObjectPos
				Object.Direction=TempValue1
			endif
		floop
		if Object[ArrayPos0].XPos<Object.XPos
			Object.XVelocity-=0x1000
			if Object.XVelocity<-0x20000
				Object.XVelocity=-0x20000
			endif
		else
			Object.XVelocity+=0x1000
			if Object.XVelocity>0x20000
				Object.XVelocity=0x20000
			endif
		endif
		if Object[ArrayPos0].YPos<Object.YPos
			Object.YVelocity-=0x1000
			if Object.YVelocity<-0x20000
				Object.YVelocity=-0x20000
			endif
		else
			Object.YVelocity+=0x1000
			if Object.YVelocity>0x20000
				Object.YVelocity=0x20000
			endif
		endif
		Object.Value0--
		if Object.Value0<0
			Rand(Object.Value0,32)
			Object.State=0
			Object.XVelocity=0
			Object.YVelocity=0
			Object.Direction=FLIP_NONE
		endif
		break
	case 2
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		break
	endswitch
	Object.AnimationTimer++
	Object.AnimationTimer&=3
	Object.Frame=Object.AnimationTimer
	Object.Frame>>=1
	Object.Frame++
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value2
		Object.YPos=Object.Value3
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(Whisp_Function145)
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-8,8,8,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub
