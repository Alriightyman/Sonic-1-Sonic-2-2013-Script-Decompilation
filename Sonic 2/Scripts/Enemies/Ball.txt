sub ObjectMain
	Object.Priority=PRIORITY_ACTIVE
	switch Object.State
	case 0
		if Object.Direction==FLIP_X
			Object.YVelocity+=0x800
			if Object.YVelocity>=0xC000
				Object.Direction=FLIP_NONE
			endif
		else
			Object.YVelocity-=0x800
			if Object.YVelocity<=-0xC000
				Object.Direction=FLIP_X
			endif
		endif
		Object.YPos+=Object.YVelocity
		foreach TypeGroup[256],PlayerObjectPos
			if Object.State==0
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-256,-192,256,192,PlayerObjectPos,0,0,0,0)
				if CheckResult==1
					Object.Value2=PlayerObjectPos
					Object.State=1
				endif
			endif
		floop
		break
	case 1
		PlayerObjectPos=Object.Value2
		if Object.XPos>Object[PlayerObjectPos].XPos
			Object.XVelocity-=0x1000
			if Object.XVelocity<-0x20000
				Object.XVelocity=-0x20000
			endif
		else
			Object.XVelocity+=0x1000
			if Object.XVelocity>0x20000
				Object.XVelocity=0x20000
			endif
		endif
		TempValue0=Object[PlayerObjectPos].YPos
		TempValue0-=0x400000
		if Object.YPos>TempValue0
			Object.YVelocity-=0x1000
			if Object.YVelocity<-0x20000
				Object.YVelocity=-0x20000
			endif
		else
			Object.YVelocity+=0x1000
			if Object.YVelocity>0x20000
				Object.YVelocity=0x20000
			endif
		endif
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		TempValue0=Object.XPos
		TempValue0-=Object[PlayerObjectPos].XPos
		Absolute(TempValue0)
		if TempValue0<=0x100000
			TempValue0=Object[PlayerObjectPos].YPos
			TempValue0-=Object.YPos
			if TempValue0>=0x400000
				if TempValue0<=0x600000
					if Object.OutOfBounds==0
						CreateTempObject(TypeName[SmokePuff],0,Object.XPos,Object.YPos)
						Object[TempObjectPos].DrawOrder=4
						CreateTempObject(TypeName[ChemicalDrop],0,Object.XPos,Object.YPos)
						Object[TempObjectPos].DrawOrder=2
						Object.Type=TypeName[BlankObject]
						PlaySfx(SFXName[Destroy],0)
					endif
				endif
			endif
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-10,-10,10,10,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				CallFunction(PlayerObject_Function45)
				CreateTempObject(TypeName[SmokePuff],0,Object.XPos,Object.YPos)
				Object[TempObjectPos].DrawOrder=4
				CreateTempObject(TypeName[ChemicalDrop],0,Object.XPos,Object.YPos)
				Object[TempObjectPos].DrawOrder=2
				Object.Type=TypeName[BlankObject]
				PlaySfx(SFXName[Destroy],0)
			endif
		floop
		break
	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value1
		Object.YPos=Object.Value2
		if Object.OutOfBounds==1
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.State=0
			Object.Value2=-1
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	Object.AnimationTimer++
	if Object.AnimationTimer>=3
		Object.AnimationTimer=0
		Object.Frame++
		if Object.Frame>=3
			Object.Frame=0
		endif
	endif
endsub


sub ObjectDraw
	DrawSprite(Object.Frame)
endsub
