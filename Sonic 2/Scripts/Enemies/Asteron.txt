function Asteron_Function108
	Object[ArrayPos0].Frame=0
	Object[ArrayPos0].Value0=0
	Object[ArrayPos0].State=0
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction


function Asteron_Function109
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Asteron_Function110
	CreateTempObject(TypeName[Asteron],0,Object.XPos,Object.YPos)
	ArrayPos0=Object[TempObjectPos].EntityNo
	Object[ArrayPos0].Value1=Object.XPos
	Object[ArrayPos0].Value2=Object.YPos
	Object[ArrayPos0].DrawOrder=5
	CallFunction(Asteron_Function108)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		TempValue0=0
		TempValue1=0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-96,-64,96,64,PlayerObjectPos,-1,-1,1,1)
			if CheckResult==0
				TempValue0++
			endif
			TempValue1++
		floop
		if TempValue0==TempValue1
			Object.Priority=PRIORITY_ACTIVE
			Object.State=1
		endif
		break
	case 1
		foreach TypeGroup[256],PlayerObjectPos
			if Object.State==1
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-96,-96,96,96,PlayerObjectPos,-1,-1,1,1)
				if CheckResult==1
					TempValue1=0
					TempValue0=Object.XPos
					TempValue0-=Object[PlayerObjectPos].XPos
					Absolute(TempValue0)
					if TempValue0>=0x100000
						TempValue1=1
						if Object.XPos>Object[PlayerObjectPos].XPos
							Object.XVelocity=-0x4000
						else
							Object.XVelocity=0x4000
						endif
					endif
					TempValue0=Object.YPos
					TempValue0-=Object[PlayerObjectPos].YPos
					Absolute(TempValue0)
					if TempValue0>=0x100000
						TempValue1=1
						if Object.YPos>Object[PlayerObjectPos].YPos
							Object.YVelocity=-0x4000
						else
							Object.YVelocity=0x4000
						endif
					endif
					if TempValue1==1
						Object.Priority=PRIORITY_ACTIVE
						Object.Value0=65
						Object.State=2
					endif
				endif
			endif
		floop
		break
	case 2
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=1
		Object.AnimationTimer++
		Object.AnimationTimer&=3
		Object.Value0--
		if Object.Value0==0
			ResetObjectEntity(Object.EntityNo,TypeName[BlankObject],0,Object.XPos,Object.YPos)
			CreateTempObject(TypeName[AsteronSpike],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].YPos-=0x80000
			Object[TempObjectPos].YVelocity=-0x40000
			Object[TempObjectPos].DrawOrder=5
			CreateTempObject(TypeName[AsteronSpike],1,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos+=0x80000
			Object[TempObjectPos].YPos-=0x40000
			Object[TempObjectPos].XVelocity=0x30000
			Object[TempObjectPos].YVelocity=-0x10000
			Object[TempObjectPos].DrawOrder=5
			CreateTempObject(TypeName[AsteronSpike],2,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos+=0x80000
			Object[TempObjectPos].YPos+=0x80000
			Object[TempObjectPos].XVelocity=0x30000
			Object[TempObjectPos].YVelocity=0x30000
			Object[TempObjectPos].DrawOrder=5
			CreateTempObject(TypeName[AsteronSpike],2,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos-=0x80000
			Object[TempObjectPos].YPos+=0x80000
			Object[TempObjectPos].XVelocity=-0x30000
			Object[TempObjectPos].YVelocity=0x30000
			Object[TempObjectPos].Direction=FLIP_X
			Object[TempObjectPos].DrawOrder=5
			CreateTempObject(TypeName[AsteronSpike],1,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos-=0x80000
			Object[TempObjectPos].YPos-=0x40000
			Object[TempObjectPos].XVelocity=-0x30000
			Object[TempObjectPos].YVelocity=-0x10000
			Object[TempObjectPos].Direction=FLIP_X
			Object[TempObjectPos].DrawOrder=5
			CreateTempObject(TypeName[SmokePuff],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].YPos+=0x20000
			Object[TempObjectPos].DrawOrder=5
			PlaySfx(SFXName[Destroy],0)
		endif
		break
	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value1
		Object.YPos=Object.Value2
		Object.YPos+=Object.Value31
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(Asteron_Function108)
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-8,8,8,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub
