function Buzzer_Function123
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)

function Buzzer_Function124
	CreateTempObject(TypeName[Buzzer],0,Object.XPos,Object.YPos)
	GetBit(TempValue0,Object.Direction,0)
	if TempValue0==0
		Object[TempObjectPos].Direction=FLIP_NONE
		Object[TempObjectPos].XVelocity=-0x10000
	else
		Object[TempObjectPos].Direction=FLIP_X
		Object[TempObjectPos].XVelocity=0x10000
	endif
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		if Object.Value0<256
			Object.Value0++
			Object.XPos+=Object.XVelocity
		else
			Object.Value0=0
			Object.State=1
			Object.Animation=1
			Object.Direction^=FLIP_X
			FlipSign(Object.XVelocity)
		endif
		break
	case 1
		if Object.Value3<30
			Object.Value3++
		else
			Object.Value3=0
			Object.State=0
			Object.Animation=0
			Object.Value2=0
		endif
		break
	case 2
		if Object.Value3<50
			Object.Value3++
			if Object.Value3==30
				CreateTempObject(TypeName[BuzzerShot],0,Object.XPos,Object.YPos)
				Object[TempObjectPos].Direction=Object.Direction
				if Object.Direction==FLIP_NONE
					Object[TempObjectPos].XPos+=0xD0000
					Object[TempObjectPos].XVelocity=-0x18000
				else
					Object[TempObjectPos].XPos-=0xD0000
					Object[TempObjectPos].XVelocity=0x18000
				endif
				Object[TempObjectPos].YPos+=0x180000
				Object[TempObjectPos].YVelocity=0x18000
			endif
		else
			Object.Value3=0
			Object.State=0
			Object.Animation=0
			Object.Value2=1
		endif
		break
	case 3
		Object.Animation=3
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		Object.XPos=Object.Value1
		if Object.Value3==1
			Object.State=3
		endif
		if Object.OutOfBounds==1
			Object.XPos=Object.Value1
			GetBit(TempValue0,Object.PropertyValue,0)
			if TempValue0==0
				Object.Direction=FLIP_NONE
				Object.XVelocity=-0x10000
			else
				Object.Direction=FLIP_X
				Object.XVelocity=0x10000
			endif
			Object.Value0=0
			Object.Animation=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.State=0
		else
			Object.XPos=TempValue0
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		if Object.State==0
			if Object.Value2==0
				if Object.Direction==FLIP_NONE
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-28,-256,-30,256,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				else
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,28,-256,30,256,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				endif
				if CheckResult==1
					Object.State=2
					Object.Animation=2
				endif
			endif
		endif
		if Object.State<3
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-24,-12,24,12,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
			if CheckResult==1
				CallFunction(PlayerObject_Function44)
			endif
		endif
	floop
	Object.Frame=Object.AnimationTimer
	Object.Frame/=3
	Object.Frame+=3
	Object.AnimationTimer++
	if Object.AnimationTimer>=6
		Object.AnimationTimer=0
	endif
endsub


sub ObjectDraw
	switch Object.Animation
	case 0
		DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 1
		DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 2
		DrawSpriteFX(1,FX_FLIP,Object.XPos,Object.YPos)
		break

	endswitch
endsub
