function BubblersMother_Function153
	Object[ArrayPos0].Direction=Object[ArrayPos0].PropertyValue
	if Object[ArrayPos0].Direction==FLIP_NONE
		Object[ArrayPos0].XVelocity=-0x4000
	else
		Object[ArrayPos0].XVelocity=0x4000
	endif
	Object[ArrayPos0].YVelocity=0
	Object[ArrayPos0].Value0=512
	Object[ArrayPos0].Value1=0
	Object[ArrayPos0].Frame=1
	Object[ArrayPos0].AnimationTimer=0
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
	Object[ArrayPos0].State=0
endfunction


function BubblersMother_Function154
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function BubblersMother_Function155
	CreateTempObject(TypeName[BubblersMother],Object.Direction,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value2=Object.XPos
	Object[TempObjectPos].Value3=Object.YPos
	ArrayPos0=Object[TempObjectPos].EntityNo
	CallFunction(BubblersMother_Function153)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.XPos+=Object.XVelocity
		Object.AnimationTimer++
		Object.AnimationTimer&=3
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=1
		Object.Frame++
		Object.Value0--
		if Object.Value0==0
			Object.Direction^=FLIP_X
			FlipSign(Object.XVelocity)
			Object.Value0=512
		endif
		foreach TypeGroup[256],PlayerObjectPos
			if Object.State==0
				if Object.Direction==FLIP_NONE
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-160,12,16,96,PlayerObjectPos,-1,-1,1,1)
				else
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,16,12,160,96,PlayerObjectPos,-1,-1,1,1)
				endif
				if CheckResult==1
					Object.State=1
					Object.Frame=2
					Object.Value0=16
					if Object.Direction==FLIP_NONE
						Object.XVelocity=-0x28000
					else
						Object.XVelocity=0x28000
					endif
				endif
			endif
		floop
		break
	case 1
		Object.Value0--
		if Object.Value0<0
			Object.State=2
		endif
		Object.AnimationTimer++
		Object.AnimationTimer&=3
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=1
		Object.Frame++
		break
	case 2
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.Value1++
		if Object.Value1>=20
			Object.Value1=0
			TempValue0=Object.XPos
			if Object.Direction==FLIP_NONE
				TempValue0+=0x60000
			else
				TempValue0-=0x60000
			endif
			TempValue1=Object.YPos
			TempValue1+=0xA0000
			CreateTempObject(TypeName[Bubbler],0,TempValue0,TempValue1)
		endif
		Object.AnimationTimer++
		if Object.AnimationTimer>=3
			Object.AnimationTimer=0
		endif
		Object.Frame=Object.AnimationTimer
		Object.Frame++
		break
	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value2
		Object.YPos=Object.Value3
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(BubblersMother_Function153)
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-12,16,12,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub


sub ObjectDraw

	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endsub
