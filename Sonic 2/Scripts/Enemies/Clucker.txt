function Clucker_Function128
	Object[ArrayPos0].YPos=Object[ArrayPos0].Value0
	Object[ArrayPos0].Value1=0
	Object[ArrayPos0].State=0
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction


function Clucker_Function129
	EditFrame(0,-16,-16,32,32,Clucker_staticVar49,Clucker_staticVar50)
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Clucker_Function130
	CreateTempObject(TypeName[Clucker],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].YPos+=0x200000
	Object[TempObjectPos].Value0=Object[TempObjectPos].YPos
	Object[TempObjectPos].Direction=Object.Direction
	ArrayPos0=Object[TempObjectPos].EntityNo
	CallFunction(Clucker_Function128)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-128,-256,128,32,PlayerObjectPos,-1,-1,1,1)
			if CheckResult==1
				Object.Value1=16
				if Object[PlayerObjectPos].XPos>Object.XPos
					Object.Direction=FLIP_X
					Object.Value2=-0x80000
				else
					Object.Direction=FLIP_NONE
					Object.Value2=0x80000
				endif
				Object.Priority=PRIORITY_ACTIVE
				Object.State=1
			endif
		floop
		break
	case 1
		Object.YPos-=0x20000
		Object.Value1--
		if Object.Value1==0
			Object.Value1=8
			Object.State=2
		endif
		break
	case 2
		if Object.Direction==FLIP_NONE
			Object.Value2-=0x10000
		else
			Object.Value2+=0x10000
		endif
		Object.Value1--
		if Object.Value1==0
			Object.Value1=14
			Object.State=3
		endif
		break
	case 3
		switch Object.Value1
		case 4
			if Object.Direction==FLIP_NONE
				Object.Value2+=0x20000
			else
				Object.Value2-=0x20000
			endif
			break
		case 0
			PlaySfx(SFXName[OctusShot],0)
			TempValue0=Object.XPos
			TempValue1=Object.YPos
			TempValue1+=0xB0000
			if Object.Direction==FLIP_NONE
				Object.Value2-=0x20000
				TempValue0-=0x130000
				CreateTempObject(TypeName[CluckerShot],0,TempValue0,TempValue1)
				Object[TempObjectPos].XVelocity=-0x20000
			else
				Object.Value2+=0x20000
				TempValue0+=0x130000
				CreateTempObject(TypeName[CluckerShot],0,TempValue0,TempValue1)
				Object[TempObjectPos].XVelocity=0x20000
			endif
			Object[TempObjectPos].Direction=Object.Direction
			Object.Value1=68
			break
		case 1
		case 2
		case 3
		endswitch
		Object.Value1--
		break
	endswitch
	TempValue0=Object.YPos
	Object.YPos=Object.Value0
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
	floop
	Object.YPos=TempValue0
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-14,-16,14,16,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
	if Object.OutOfBounds==1
		TempValue1=Object.YPos
		Object.YPos=Object.Value0
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(Clucker_Function128)
		else
			Object.YPos=TempValue1
		endif
	endif
endsub


sub ObjectDraw
	if Object.State>=2
		TempValue0=Object.XPos
		TempValue0+=Object.Value2
		DrawSpriteFX(1,FX_FLIP,TempValue0,Object.YPos)
	endif
	TempValue0=Object.Value0
	TempValue0-=Object.YPos
	TempValue0>>=16
	EditFrame(0,-16,-16,32,TempValue0,Clucker_staticVar49,Clucker_staticVar50)
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endsub
