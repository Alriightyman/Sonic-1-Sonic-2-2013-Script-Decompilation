sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.Value7=0x7FFFFFFF
		Object.Value5=0
		Object.Value6=0xFFFF
		foreach TypeGroup[256],PlayerObjectPos
			TempValue0=Object[PlayerObjectPos].XPos
			TempValue0-=Object.XPos
			if TempValue0<0
				TempValue1=0
				FlipSign(TempValue0)
			else
				TempValue1=1
			endif
			if TempValue0<Object.Value7
				Object.Value7=TempValue0
				Object.Value5=PlayerObjectPos
				Object.Direction=TempValue1
			endif
		floop
		TempValue0=0
		if Object.Value1==0
			if Object.Value7>-0x600000
				if Object.Value7<0x600000
					Object.Value0=8
					Object.Value1=32
					Object.Frame=1
					Object.State=2
					TempValue0=1
				endif
			endif
		else
			Object.Value1--
		endif
		if TempValue0==0
			Object.Value0--
			if Object.Value0<0
				GetBit(TempValue0,Object.Value2,0)
				if TempValue0==0
					Object.YVelocity=-0x10000
				else
					Object.YVelocity=0x10000
				endif
				GetArrayValue(Object.Value0,Object.Value2,Coconuts_array46)
				Object.Value2++
				if Object.Value2>5
					Object.Value2=0
				endif
				Object.State=1
			endif
		endif
		break
	case 1
		Object.AnimationTimer++
		if Object.AnimationTimer==6
			Object.Frame=1
		endif
		if Object.AnimationTimer==12
			Object.Frame=0
			Object.AnimationTimer=0
		endif
		Object.YPos+=Object.YVelocity
		Object.Value0--
		if Object.Value0<0
			Object.Value0=16
			Object.State=0
		endif
		break
	case 2
		Object.Value0--
		if Object.Value0<0
			CreateTempObject(TypeName[Coconut],0,Object.XPos,Object.YPos)
			if Object.Direction==FLIP_NONE
				Object[TempObjectPos].XPos+=0xB0000
				Object[TempObjectPos].XVelocity=-0x10000
			else
				Object[TempObjectPos].XPos-=0xB0000
				Object[TempObjectPos].XVelocity=0x10000
			endif
			Object[TempObjectPos].YPos-=0xD0000
			Object[TempObjectPos].YVelocity=-0x10000
			Object[TempObjectPos].Direction=Object.Direction
			Object[TempObjectPos].Priority=PRIORITY_ACTIVE
			Object.Value0=8
			Object.Frame=2
			Object.State=3
		endif
		break
	case 3
		Object.Value0--
		if Object.Value0<0
			GetBit(TempValue0,Object.Value2,0)
			if TempValue0==0
				Object.YVelocity=-0x10000
			else
				Object.YVelocity=0x10000
			endif
			GetArrayValue(Object.Value0,Object.Value2,Coconuts_array46)
			Object.Value2++
			if Object.Value2>5
				Object.Value2=0
			endif
			Object.Frame=0
			Object.State=1
		endif
		break
	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value3
		Object.YPos=Object.Value4
		if Object.OutOfBounds==1
			Object.Value1=0
			Object.Value0=16
			Object.State=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-12,-16,12,16,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub


sub ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
endsub
