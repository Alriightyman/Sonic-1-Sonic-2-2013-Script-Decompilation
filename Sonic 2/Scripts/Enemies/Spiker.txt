#function Spiker_Function121
#function Spiker_Function122
#function Spiker_Function123



function Spiker_Function121
	Object[ArrayPos0].Direction=Object[ArrayPos0].PropertyValue
	GetBit(TempValue0,Object[ArrayPos0].Direction,0)
	if TempValue0==0
		Object[ArrayPos0].XVelocity=-0x8000
	else
		Object[ArrayPos0].XVelocity=0x8000
	endif
	Object[ArrayPos0].YVelocity=0
	Object[ArrayPos0].Value0=64
	Object[ArrayPos0].Frame=0
	Object[ArrayPos0].Animation=0
	Object[ArrayPos0].AnimationTimer=0
	Object[ArrayPos0].Value2=1
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction


function Spiker_Function122
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
	DrawSpriteFX(2,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Spiker_Function123
	CreateTempObject(TypeName[Spiker],Object.Direction,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value1=Object.XPos
	ArrayPos0=Object[TempObjectPos].EntityNo
	CallFunction(Spiker_Function121)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.XPos+=Object.XVelocity
		Object.Value0--
		if Object.Value0==0
			Object.Value0=16
			Object.State=1
		endif
		Object.Frame=Object.AnimationTimer
		Object.Frame/=10
		Object.AnimationTimer++
		if Object.AnimationTimer==20
			Object.AnimationTimer=0
		endif
		break
	case 1
		Object.Value0--
		if Object.Value0<0
			Object.Value0=64
			Object.Direction^=FLIP_X
			FlipSign(Object.XVelocity)
			Object.State=0
		endif
		break
	case 2
		Object.Value3--
		if Object.Value3<0
			Object.Value2=0
			CreateTempObject(TypeName[Spiker],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].Value2=1
			Object[TempObjectPos].State=3
			Object[TempObjectPos].Direction=Object.Direction
			GetBit(TempValue0,Object.Direction,1)
			if TempValue0==0
				Object[TempObjectPos].YVelocity=-0x20000
			else
				Object[TempObjectPos].YVelocity=0x20000
			endif
			Object.State=Object.Value4
		endif
		break
	case 3
		Object.YPos+=Object.YVelocity
		Object.Direction^=FLIP_X
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		Object.XPos=Object.Value1
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(Spiker_Function121)
		else
			Object.XPos=TempValue0
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		if Object.State!=3
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-10,8,10,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
			if CheckResult==1
				CallFunction(PlayerObject_Function44)
			endif
			if Object.Type!=TypeName[BlankObject]
				if Object.Value2==1
					PlayerObjectCollision(C_TOUCH,Object.EntityNo,-32,-128,32,128,PlayerObjectPos,-1,-1,1,1)
					if CheckResult==1
						Object.Value2=2
						Object.Value3=8
						Object.Value4=Object.State
						Object.State=2
					endif
				endif
			endif
		endif
		if Object.Value2>=1
			GetBit(TempValue0,Object.Direction,1)
			if TempValue0==0
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-32,8,0,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
			else
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,0,8,32,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
			endif
			if CheckResult==1
				CallFunction(PlayerObject_Function45)
			endif
		endif
	floop
endsub


sub ObjectDraw
	if Object.State!=3
		DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
	endif
	if Object.Value2>=1
		DrawSpriteFX(2,FX_FLIP,Object.XPos,Object.YPos)
	endif
endsub

	CheckStageFolder("Zone05")
	if CheckResult==1
		LoadSpriteSheet("HTZ/Objects.gif")

		SpriteFrame(-12,-8,24,24,66,231)

		SpriteFrame(-12,-8,24,24,67,806)
		SpriteFrame(-12,-32,24,32,67,748)
	endif
	foreach TypeName[Spiker],ArrayPos0
		Object[ArrayPos0].Value1=Object[ArrayPos0].XPos
		GetBit(TempValue0,Object[ArrayPos0].PropertyValue,1)
		if TempValue0==0
			Object[ArrayPos0].YPos+=0x80000
		else
			Object[ArrayPos0].YPos-=0x80000
		endif
		CallFunction(Spiker_Function121)
	floop
	SetArrayValue(51,DebugMode_constant25,DebugMode_array12)
