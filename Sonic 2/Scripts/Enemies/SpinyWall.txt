function SpinyWall_Function147
	Object[ArrayPos0].XPos=Object[ArrayPos0].Value0
	Object[ArrayPos0].YPos=Object[ArrayPos0].Value1
	Object[ArrayPos0].Value2=256
	Object[ArrayPos0].Value3=128
	Object[ArrayPos0].Frame=0
	Object[ArrayPos0].AnimationTimer=0
	Object[ArrayPos0].State=0
	Object[ArrayPos0].Direction=Object[ArrayPos0].PropertyValue
	Object[ArrayPos0].YVelocity=-0x4000
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction

endfunction


function SpinyWall_Function149
	CreateTempObject(TypeName[SpinyWall],TempValue0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value0=Object.XPos
	Object[TempObjectPos].Value1=Object.YPos
	ArrayPos0=Object[TempObjectPos].EntityNo
	Object[TempObjectPos].PropertyValue=Object.Direction

endfunction


sub ObjectMain
	Object.Priority=PRIORITY_ACTIVE
	switch Object.State
	case 0
		Object.Value2--
		if Object.Value2<=0
			Object.Value2=256
			Object.Value3=128
			FlipSign(Object.YVelocity)
			Object.Direction^=FLIP_Y
		endif
		Object.YPos+=Object.YVelocity
		Object.AnimationTimer++
		if Object.AnimationTimer==10
			Object.AnimationTimer=0
			Object.Frame^=1
		endif
		if Object.Value3==0
			Object.Value5=0x7FFFFFFF
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-96,-256,96,256,PlayerObjectPos,1,1,1,1)
				if CheckResult==1
					Object.Value3=40
					Object.Frame=2
					Object.State=1
					TempValue0=Object.XPos
					TempValue0-=Object[PlayerObjectPos].XPos
					Absolute(TempValue0)
					if TempValue0<Object.Value5
						Object.Value5=TempValue0
						if Object[PlayerObjectPos].XPos>Object.XPos
							Object.Value4=0x30000
						else
							Object.Value4=-0x30000
						endif
					endif
				endif
			floop
		else
			Object.Value3--
		endif
		break
	case 1
		Object.Value3--
		if Object.Value3==20
			CreateTempObject(TypeName[SpinyShot],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].XVelocity=Object.Value4
			Object[TempObjectPos].DrawOrder=2
		endif
		if Object.Value3<0
			Object.Value3=64
			Object.Frame=0
			Object.State=0
		endif
		break
	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value0
		Object.YPos=Object.Value1
		if Object.OutOfBounds==1
			ArrayPos0=Object.EntityNo
			CallFunction(SpinyWall_Function147)
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-8,-8,8,8,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function44)
		endif
	floop
endsub


sub ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
endsub
