#alias 73: TYPE_EGGMANMINE




sub ObjectMain
	switch Object.State
	case 0
		if Object.PropertyValue==0
			Object.Value1=-12
			Object.Value2=12
			Object.Value3=-12
			Object.Value4=12
		else
			Object.Value1=-16
			Object.Value2=16
			Object.Value3=-24
			Object.Value4=16
		endif
		Object.State++
		break
	case 1
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x1800
		TempValue0=Stage.WaterLevel
		TempValue0<<=16
		if Object.YPos>TempValue0
			CreateTempObject(TypeName[WaterSplash],0,Object.XPos,TempValue0)
			PlaySfx(SFXName[WaterSplash],0)
			Object[TempObjectPos].DrawOrder=4
			Object.YVelocity>>=2
			if Object.PropertyValue==0
				Sin(TempValue1,oscillation)
				TempValue1>>=5
				TempValue1<<=16
				Object.Value5=TempValue1
				Object.Value5+=Object.XPos
				Object.State++
			else
				Object.State=5
			endif
		endif
		break
	case 2
		Object.YPos+=Object.YVelocity
		Object.YVelocity-=0x1800
		TempValue0=Stage.WaterLevel
		TempValue0<<=16
		if Object.YPos<=TempValue0
			Object.YPos=TempValue0
			Object.YVelocity=0
			Object.State++
		endif
		Sin(Object.XPos,oscillation)
		Object.XPos>>=5
		Object.XPos<<=16
		FlipSign(Object.XPos)
		Object.XPos+=Object.Value5
		break
	case 3
		Object.YPos=Stage.WaterLevel
		Object.YPos<<=16
		Sin(Object.XPos,oscillation)
		Object.XPos>>=5
		Object.XPos<<=16
		FlipSign(Object.XPos)
		Object.XPos+=Object.Value5
		break
	case 4
		Object.Value0++
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=5
		Object.AnimationTimer+=Object.Value0
		Object.AnimationTimer&=63
		if Object.Value0==60
			Object.Value3-=16
			Object.Type=TypeName[BlankObject]
			CreateTempObject(TypeName[Explosion],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			CreateTempObject(TypeName[Explosion],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].YPos-=0x180000
			Object[TempObjectPos].DrawOrder=4
			CreateTempObject(TypeName[WaterSplash],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			PlaySfx(SFXName[Explosion],0)
		endif
		break
	case 5
		Object.YPos+=Object.YVelocity
		Object.YVelocity-=0x1000
		TempValue0=Stage.WaterLevel
		TempValue0<<=16
		if Object.YPos<=TempValue0
			Object.YPos=TempValue0
			Object.YVelocity=0
			Object.State++
		endif
		break
	case 6
		Object.YPos=Stage.WaterLevel
		Object.YPos<<=16
		TempValue0=Object[0].XPos
		TempValue0-=0x20000
		TempValue1=Object[0].XPos
		TempValue1+=0x20000
		if Object.XPos<TempValue0
			Object.XPos+=0x8000
			Object.Direction=FLIP_X
		endif
		if Object.XPos>TempValue1
			Object.XPos-=0x8000
			Object.Direction=FLIP_NONE
		endif
		Object.Value0++
		if Object.Value0==180
			Object.Value0=0
			Object.State++
		endif
		break
	case 7
		TempValue0=Object[0].XPos
		TempValue0-=0x20000
		TempValue1=Object[0].XPos
		TempValue1+=0x20000
		if Object.XPos<TempValue0
			Object.XPos+=0x8000
			Object.Direction=FLIP_X
		endif
		if Object.XPos>TempValue1
			Object.XPos-=0x8000
			Object.Direction=FLIP_NONE
		endif
		Object.Value0++
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=5
		Object.AnimationTimer+=Object.Value0
		Object.AnimationTimer&=63
		if Object.Value0==60
			Object.Value0=0
			Object.State++
			Object.Value1=-32
			Object.Value2=32
			ArrayPos0=Object.EntityNo
			ArrayPos0--
			ResetObjectEntity(ArrayPos0,TypeName[WaterGeyser],1,Object.XPos,Object.YPos)
			Object[ArrayPos0].DrawOrder=4
			CreateTempObject(TypeName[Explosion],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			CreateTempObject(TypeName[WaterSplash],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			PlaySfx(SFXName[BigExplosion],0)
		endif
		break
	case 8
		TempValue0=Object.Value0
		TempValue0%=6
		if TempValue0==5
			TempValue1=Object.XPos
			TempValue1-=0x180000
			TempValue2=Object.XPos
			TempValue2+=0x180000
			CreateTempObject(TypeName[SmokePuff],0,TempValue1,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			CreateTempObject(TypeName[SmokePuff],0,TempValue2,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
			Object.YPos-=0x180000
			Object.Value3-=24
			CreateTempObject(TypeName[Explosion],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder=4
		endif
		Object.Value0++
		if Object.Value0==24
			Object.Type=TypeName[BlankObject]
		endif
		break
	endswitch
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,Object.Value1,Object.Value3,Object.Value2,Object.Value4,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		if CheckResult==1
			CallFunction(PlayerObject_Function45)
		endif
	floop
	if Object.OutOfBounds==1
		Object.Type=TypeName[BlankObject]
	endif
endsub


sub ObjectDraw
	if Object.State<8
		TempValue0=Object.PropertyValue
		TempValue0+=Object.Frame
		DrawSprite(TempValue0)
	endif
endsub
