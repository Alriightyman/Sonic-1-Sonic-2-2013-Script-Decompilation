sub ObjectMain
	if Object.State>0
		Object.Frame=Object.Value0
		Object.Frame/=5
		Object.Frame++
		Object.Value0++
		if Object.Value0>15
			Object.Value0=0
			Object.State=0
			Object.Frame=0
		endif
	endif
	if Object.OutOfBounds==1
		Object.Priority=PRIORITY_ACTIVE_BOUNDS
	endif
	if Object.XVelocity!=0
		Object.XPos+=Object.XVelocity
		if Object.XVelocity>0
			if Object.XPos>=Object.Value4
				Object.XPos=Object.Value4
				FlipSign(Object.XVelocity)
			endif
		else
			if Object.XPos<=Object.Value3
				Object.XPos=Object.Value3
				FlipSign(Object.XVelocity)
			endif
		endif
	endif
	TempValue7=0
	foreach TypeGroup[256],PlayerObjectPos
		TempValue6=Object.Value5
		TempValue6>>=TempValue7
		TempValue6&=255
		if TempValue6==0
			if Object[PlayerObjectPos].Animation!=ANI_HURT
				PlayerObjectCollision(C_TOUCH,Object.EntityNo,-12,-8,12,8,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
				if CheckResult==1
					if Object.State==0
						PlaySfx(SFXName[Bumper],0)
					endif
					if Object.Value0>5
						PlaySfx(SFXName[Bumper],0)
					endif
					Object.State=1
					Object.Priority=PRIORITY_ACTIVE
					TempValue0=Object[PlayerObjectPos].XPos
					TempValue0-=Object.XPos
					TempValue1=Object[PlayerObjectPos].YPos
					TempValue1-=Object.YPos
					ATan2(TempValue2,TempValue0,TempValue1)
					TempValue2+=32
					TempValue2&=255
					TempValue2>>=6
					switch TempValue2
					case 0
						TempValue0=0x80000
						TempValue1=Object[PlayerObjectPos].YVelocity
						Object.Value1=-0x40000
						Object.Value2=0
						break
					case 1
						if Object[PlayerObjectPos].XPos<Object.XPos
							TempValue0=-0x20000
						else
							TempValue0=0x20000
						endif
						TempValue1=0x80000
						Object.Value1=0
						Object.Value2=-0x40000
						break
					case 2
						TempValue0=-0x80000
						TempValue1=Object[PlayerObjectPos].YVelocity
						Object.Value1=0x40000
						Object.Value2=0
						break
					case 3
						if Object[PlayerObjectPos].XPos<Object.XPos
							TempValue0=-0x20000
						else
							TempValue0=0x20000
						endif
						TempValue1=-0x80000
						Object.Value1=0
						Object.Value2=0x40000
						break
					endswitch
					if Object[PlayerObjectPos].State==34
						Object[1].Value36=30
					endif
					CheckEqual(Object[PlayerObjectPos].Animation,ANI_GLIDING)
					TempValue2=CheckResult
					CheckEqual(Object[PlayerObjectPos].Animation,ANI_GLIDING_STOP)
					TempValue2|=CheckResult
					CheckEqual(Object[PlayerObjectPos].Animation,ANI_FLYING)
					TempValue2|=CheckResult
					CheckEqual(Object[PlayerObjectPos].State,31)
					TempValue2|=CheckResult
					CheckEqual(Object[PlayerObjectPos].State,13)
					TempValue2|=CheckResult
					if TempValue2==0
						Object[PlayerObjectPos].State=12
						if Object[PlayerObjectPos].Animation!=ANI_JUMPING
							if Object[PlayerObjectPos].Animation!=ANI_RUNNING
								Object[PlayerObjectPos].Animation=ANI_WALKING
							endif
						endif
					endif

						Object[PlayerObjectPos].XVelocity+=TempValue0
						Object[PlayerObjectPos].Speed=Object[PlayerObjectPos].XVelocity
						Object[PlayerObjectPos].Value1=0

					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
					Object[PlayerObjectPos].TileCollisions=1
					TempValue6=3
					TempValue6<<=TempValue7
					Object.Value5|=TempValue6
				endif
			endif
		else
			TempValue5=1
			TempValue5<<=TempValue7
			Object.Value5-=TempValue5
		endif
		TempValue7+=8
	floop
endsub


sub ObjectDraw
	TempValue0=Object.Frame
	TempValue0&=1
	if TempValue0==0
		DrawSprite(0)
	else
		TempValue1=Object.XPos
		TempValue1+=Object.Value1
		TempValue2=Object.YPos
		TempValue2+=Object.Value2
		DrawSpriteXY(0,TempValue1,TempValue2)
	endif
endsub
