function Elevator_Function128
	CreateTempObject(TypeName[Elevator],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value1=Object.YPos
	Object[TempObjectPos].Value2-=0x200000
	Object[TempObjectPos].Value4-=0x400000

sub ObjectMain
	TempValue6=Object.YPos
	switch Object.State
	case 0
		if Object.Value6==1
			Object.State=1
			if Object.Priority!=PRIORITY_XBOUNDS_DESTROY
				Object.Priority=PRIORITY_ACTIVE
			endif
			PlaySfx(SFXName[Elevator],0)
		else
			foreach TypeGroup[52],ArrayPos0
				if Object.EntityNo!=ArrayPos0
					if Object[ArrayPos0].State==1
						PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-8,16,8,ArrayPos0,-16,-56,16,8)
						if CheckResult==1
							Object.State=4
							Object.Value7=0
						endif
					endif
				endif
			floop
		endif
		break
	case 1
		if Object.YPos>=Object.Value2
			Object.YVelocity-=0x800
		else
			Object.YVelocity+=0x800
		endif
		Object.YPos+=Object.YVelocity
		if Object.PropertyValue==0
			if Object.YPos<=Object.Value4
				Object.YPos=Object.Value4
				Object.YVelocity=0
				Object.State=2
			endif
		else
			if Object.YPos>=Object.Value4
				Object.YPos=Object.Value4
				Object.YVelocity=0
				Object.State=2
			endif
		endif
		break
	case 2
		if Object.Value6==0
			PlaySfx(SFXName[Elevator],0)
			Object.State=3
		endif
		break
	case 3
		if Object.YPos>=Object.Value3
			Object.YVelocity-=0x800
		else
			Object.YVelocity+=0x800
		endif
		Object.YPos+=Object.YVelocity
		if Object.PropertyValue==0
			if Object.YPos>=Object.Value1
				Object.YPos=Object.Value1
				Object.YVelocity=0
				Object.State=0
			endif
		else
			if Object.YPos<=Object.Value1
				Object.YPos=Object.Value1
				Object.YVelocity=0
				Object.State=0
			endif
		endif
		break
	case 4
		if Object.Value7==1
			Object.State=0
		endif
		Object.Value7=1
		break
	endswitch
	Object.Value6=0
	TempValue0=Object.YPos
	TempValue6&=-0x10000
	Object.Value5=Object.YPos
	Object.Value5&=-0x10000
	Object.Value5-=TempValue6
	Object.YPos=TempValue6
	foreach TypeGroup[256],PlayerObjectPos
		switch Object.State
		case 0
		case 1
		case 2
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-48,-64,48,8,PlayerObjectPos,0,0x10000,0,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].CollisionPlane=PATH_A
			endif
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-20,-32,20,8,PlayerObjectPos,0,0x10000,0,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].CollisionPlane=PATH_B
			endif
			PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0,0x10000,0,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].YPos+=Object.Value5
				Object[PlayerObjectPos].CollisionPlane=PATH_B
				Object.Value6=1
			endif
			break
		case 3
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-48,-64,48,8,PlayerObjectPos,0,0x10000,0,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].CollisionPlane=PATH_A
			endif
		case 4
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-20,-32,20,8,PlayerObjectPos,0,0x10000,0,0x10000)
			if CheckResult==1
				Object.Value7=0
			endif
		endswitch
	floop
	Object.YPos=TempValue0
	if Object.OutOfBounds==1
		TempValue0=Object.YPos
		Object.YPos=Object.Value1
		if Object.OutOfBounds==1
			Object.State=0
			if Object.Priority!=PRIORITY_XBOUNDS_DESTROY
				Object.Priority=PRIORITY_ACTIVE_BOUNDS
			endif
			Object.Value6=0
			Object.YVelocity=0
		else
			Object.YPos=TempValue0
		endif
	endif
endsub

			Object[ArrayPos0].Value4-=TempValue0
		endif
