sub ObjectMain
	TempValue0=Object.YPos
	TempValue0&=-0x10000
	switch Object.State
	case 0
		if Object.PropertyValue==0
			if Object.Value0!=0
				Object.Value0--
				if Object.Value0==0
					Object.State++
					Object.Value0=32
				endif
			endif
		endif
		break
	case 1
		Object.Value0--
		if Object.Value0==0
			Object.State++
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_BOX,Object.EntityNo,-32,-12,32,12,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if CheckResult==1
					Object[PlayerObjectPos].YPos+=Object.YVelocity
					Object[PlayerObjectPos].YVelocity=Object.YVelocity
					Object[PlayerObjectPos].YVelocity-=0x3800
					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
				endif
			floop
		endif
	case 2
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.OutOfBounds==1
			Object.YPos=Object.Value1
			Object.YPos+=Object.Value31
			Object.YVelocity=0
			Object.Value0=0
			Object.State++
		endif
		break
	case 3
		if Object.OutOfBounds==1
			Object.State=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
		endif
		break
	endswitch
	TempValue7=Object.YPos
	TempValue7&=-0x10000
	TempValue7-=TempValue0
	if Object.State<2
		TempValue2=0
		foreach TypeGroup[256],PlayerObjectPos
			GetBit(TempValue0,Object.Value2,TempValue2)
			if TempValue0==1
				Object[PlayerObjectPos].YPos+=TempValue7
			endif
			SetBit(Object.Value2,TempValue2,0)
			PlayerObjectCollision(C_BOX,Object.EntityNo,-32,-12,32,12,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			switch CheckResult
			case 1
				SetBit(Object.Value2,TempValue2,1)
				if Object.Value0==0
					if Object.State==0
						Object.Priority=PRIORITY_ACTIVE
						Object.Value0=30
					endif
				endif
				break
			case 4
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					CallFunction(PlayerObject_Function50)
				endif
				break
			case 2
			case 3
			endswitch
			TempValue2++
		floop
	endif
endsub
