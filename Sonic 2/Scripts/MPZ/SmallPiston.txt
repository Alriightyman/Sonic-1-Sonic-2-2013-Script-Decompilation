function SmallPiston_Function140
	Object[ArrayPos0].YPos=Object[ArrayPos0].Value1
	Object[ArrayPos0].YPos+=Object[ArrayPos0].Value31
	Object[ArrayPos0].Value0=90
	Object[ArrayPos0].Value2=0
	Object[ArrayPos0].State=0
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE_BOUNDS
endfunction


function SmallPiston_Function141
	DrawSprite(0)
endfunction


function SmallPiston_Function142
	CreateTempObject(TypeName[SmallPiston],0,Object.XPos,Object.YPos)
	ArrayPos0=Object[TempObjectPos].EntityNo
	Object[ArrayPos0].Value1=Object.YPos
	CallFunction(SmallPiston_Function140)
endfunction


sub ObjectMain
	TempValue0=Object.YPos
	switch Object.State
	case 0
		Object.Value0--
		if Object.Value0==0
			Object.Priority=PRIORITY_ACTIVE
			Object.Value0=8
			Object.State=1
		endif
		break
	case 1
		Object.YPos+=0x80000
		Object.Value0--
		if Object.Value0==0
			Object.Value0=90
			Object.State=2
		endif
		break
	case 2
		Object.Value0--
		if Object.Value0==0
			Object.Value0=8
			Object.State=3
		endif
		break
	case 3
		Object.YPos-=0x80000
		Object.Value0--
		if Object.Value0==0
			Object.Value0=90
			Object.State=0
		endif
		break
	endswitch
	TempValue1=Object.YPos
	TempValue1-=TempValue0
	TempValue2=0
	foreach TypeGroup[256],PlayerObjectPos
		GetBit(TempValue0,Object.Value2,TempValue2)
		if TempValue0==1
			Object[PlayerObjectPos].YPos+=TempValue1
		endif
		SetBit(Object.Value2,TempValue2,0)
		PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-32,16,32,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		switch CheckResult
		case 1
			SetBit(Object.Value2,TempValue2,1)
			break
		case 4
			if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
				CallFunction(PlayerObject_Function50)
			endif
			break
		case 2
		case 3
		endswitch
		TempValue2++
	floop
	if Object.OutOfBounds==1
		ArrayPos0=Object.EntityNo
		CallFunction(SmallPiston_Function140)
	endif
endsub


sub ObjectDraw
	DrawSprite(0)
endsub

		CallFunction(SmallPiston_Function140)
	floop
	SetArrayValue(64,DebugMode_staticVar25,DebugMode_array12)
	SetArrayValue(141,DebugMode_staticVar25,DebugMode_array13)
	SetArrayValue(142,DebugMode_staticVar25,DebugMode_array11)
