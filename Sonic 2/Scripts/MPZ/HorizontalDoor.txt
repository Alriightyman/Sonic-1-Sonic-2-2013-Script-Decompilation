function HorizontalDoor_Function166
	Object[ArrayPos0].Priority=PRIORITY_ACTIVE
	Object[ArrayPos0].Value0=0
	Object[ArrayPos0].XPos=Object[ArrayPos0].Value1
	if Object[ArrayPos0].Direction==FLIP_X
		Object[ArrayPos0].XPos-=0x400000
	endif
endfunction


function HorizontalDoor_Function167
	TempValue1=Object.XPos
	if Object.Direction==FLIP_X
		TempValue1-=0x400000
	endif
	DrawSpriteXY(0,TempValue1,Object.YPos)
endfunction


function HorizontalDoor_Function168
	CreateTempObject(TypeName[HorizontalDoor],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Direction=Object.Direction
	ArrayPos0=Object[TempObjectPos].EntityNo
	Object[ArrayPos0].Value1=Object.XPos
	CallFunction(HorizontalDoor_Function166)
endfunction


sub ObjectMain
	TempValue0=0
	TempValue1=Object.XPos
	Object.XPos=Object.Value1
	foreach TypeGroup[256],PlayerObjectPos
		if Object.Direction==FLIP_NONE
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-32,-16,96,64,PlayerObjectPos,-1,-1,1,1)
			if CheckResult==1
				TempValue0=1
			endif
		else
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-160,-16,-32,64,PlayerObjectPos,-1,-1,1,1)
			if CheckResult==1
				TempValue0=1
			endif
		endif
	floop
	Object.XPos=TempValue1
	if TempValue0==1
		if Object.Value0<4
			Object.Value0++
			if Object.Direction==FLIP_NONE
				Object.XPos-=0x100000
			else
				Object.XPos+=0x100000
			endif
		endif
	else
		if Object.Value0>0
			Object.Value0--
			if Object.Direction==FLIP_NONE
				Object.XPos+=0x100000
			else
				Object.XPos-=0x100000
			endif
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		if Object[PlayerObjectPos].YVelocity>=0
			PlayerObjectCollision(C_BOX,Object.EntityNo,-32,-12,32,12,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		endif
	floop
endsub

endsub
