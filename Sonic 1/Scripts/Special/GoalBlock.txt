sub ObjectMain
	foreach TypeGroup[256],PlayerObjectPos
		CallFunction(SpecialSetup_Function10)

			Object[PlayerObjectPos].State=1
			Object[PlayerObjectPos].XVelocity=0
			Object[PlayerObjectPos].YVelocity=0
			Object[PlayerObjectPos].Speed=0
			Object.Value0=0
			Object[PlayerObjectPos].ObjectInteractions=0

			Object[20].Priority=PRIORITY_ACTIVE
			PlaySfx(SFXName[ExitSS],0)
			Stage.TimeEnabled=0
			options.touchControls=0
		endif
	floop
endsub
