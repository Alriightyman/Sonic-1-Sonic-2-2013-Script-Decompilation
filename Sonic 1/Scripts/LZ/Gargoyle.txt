#alias 53: TYPE_GARGOYLE

// Function declarations
#function Gargoyle_DebugDraw
#function Gargoyle_DebugSetup



function Gargoyle_DebugDraw
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Gargoyle_DebugSetup
	CreateTempObject(TypeName[Gargoyle],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Direction=Object.Direction
	Object[TempObjectPos].Value1=60
	Object[TempObjectPos].Value0=60
endfunction


sub ObjectMain
	TempValue0=Object.iXPos
	TempValue0-=Screen.CameraXPos
	Absolute(TempValue0)
	if TempValue0<Screen.CenterX
		TempValue0=Object.iYPos
		TempValue0-=Screen.CameraYPos
		Absolute(TempValue0)
		if TempValue0<Screen.CenterY
			Object.Value0--
			if Object.Value0<=0
				PlaySfx(SFXName[SmallFireball],0)
				CreateTempObject(TypeName[GargoyleShot],0,Object.XPos,Object.YPos)
				Object[TempObjectPos].DrawOrder=2
				if Object.Direction==FLIP_NONE
					Object[TempObjectPos].XVelocity=-0x20000
				else
					Object[TempObjectPos].XVelocity=0x20000
				endif
				Object[TempObjectPos].YPos+=0x80000
				Object[TempObjectPos].Direction=Object.Direction
				Object.Value0=Object.Value1
			endif
		endif
	endif
endsub


sub ObjectDraw
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endsub


sub ObjectStartup
	LoadSpriteSheet("LZ/Objects.gif")
	SpriteFrame(-16,-15,32,31,93,128)
	foreach TypeName[Gargoyle],ArrayPos0
		Object[ArrayPos0].Value1=Object[ArrayPos0].PropertyValue
		Object[ArrayPos0].Value1++
		Object[ArrayPos0].Value1*=30
		Object[ArrayPos0].Value0=Object[ArrayPos0].Value1
	floop
	SetArrayValue(TypeName[Gargoyle],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(Gargoyle_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(Gargoyle_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
