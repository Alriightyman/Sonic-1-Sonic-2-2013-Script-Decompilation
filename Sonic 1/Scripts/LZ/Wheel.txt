sub ObjectMain
	if Object.Value0!=LZSetup_staticVar29
		Object.Direction^=FLIP_X
		Object.Value0=LZSetup_staticVar29
		TempValue2=1
	else
		Object.Frame=oscillation
		Object.Frame>>=2
		Object.Frame&=3
		if Object.Direction==FLIP_X
			Object.Frame^=3
		endif
	endif
endsub


sub ObjectDraw
	DrawSprite(Object.Frame)
endsub
