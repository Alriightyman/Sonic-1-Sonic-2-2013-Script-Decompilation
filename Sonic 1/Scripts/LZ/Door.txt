#alias 62: TYPE_DOOR

// Function declarations
#function Door_DebugDraw
#function Door_DebugSpawn



function Door_DebugDraw
	DrawSprite(0)
endfunction


function Door_DebugSpawn
	CreateTempObject(TypeName[Door],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Priority=PRIORITY_ACTIVE
	Object[TempObjectPos].Value0=32
endfunction


sub ObjectMain
	switch Object.State
	case 0
		if Object.PropertyValue==2
			if Object[0].XPos<Object.XPos
				CurrentTunnel_staticVar33=4
			endif
		endif
		if Object[+1].Value0==1
			Object.State++
		endif
		break
	case 1
		Object.YPos-=0x20000
		Object.Value0--
		if Object.Value0<0
			Object.State++
			if Object.PropertyValue==1
				Object.State++
			endif
		endif
		break
	case 2
		break
	case 3
		if Object[0].XPos>0x11200000
			Object.State++
		endif
		break
	case 4
		Object.YPos+=0x20000
		Object.Value0++
		if Object.Value0>=32
			Object.State=0
		endif
		break

	endswitch
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_BOX,Object.EntityNo,-8,-32,8,32,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
	floop
endsub


sub ObjectDraw
	DrawSprite(0)
endsub


sub ObjectStartup
	LoadSpriteSheet("LZ/Objects.gif")
	SpriteFrame(-8,-32,16,64,206,142)
	foreach TypeName[Door],ArrayPos0
		Object[ArrayPos0].Priority=PRIORITY_ACTIVE
		Object[ArrayPos0].Value0=32
	floop
	SetArrayValue(TypeName[Door],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(Door_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(Door_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
