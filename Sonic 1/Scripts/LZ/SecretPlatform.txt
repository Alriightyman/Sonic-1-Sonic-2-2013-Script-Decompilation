#alias 61: TYPE_MOVINGBLOCK

// Function declarations
#function MovingBlock_DebugDraw
#function MovingBlock_DebugSpawn



function MovingBlock_DebugDraw
	DrawSprite(0)
endfunction


function MovingBlock_DebugSpawn
	CreateTempObject(TypeName[MovingBlock],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Priority=PRIORITY_XBOUNDS
	Object[TempObjectPos].State=1
endfunction


sub ObjectMain
	switch Object.State
	case 0
		if Object[+2].Value0==1
			Object.State++
		endif
		break
	case 1
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object.State++
				if Stage.DebugMode==0
					if PlayerObjectPos==0
						EngineCallback(SetAchievement,5,100)
					endif
				endif
			endif
		floop
		break
	case 2
		TempValue0=Object.XPos
		TempValue0&=-0x10000
		Object.XPos+=0x10000
		ObjectTileCollision(CMODE_LWALL,16,0,0)
		if CheckResult==1
			Object.State++
		endif
		Object.Value0=Object.XPos
		Object.Value0&=-0x10000
		Object.Value0-=TempValue0
		TempValue1=Object.XPos
		Object.XPos=TempValue0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].XPos+=Object.Value0
			endif
		floop
		Object.XPos=TempValue1
		break
	case 3
		TempValue0=Object.YPos
		TempValue0&=-0x10000
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		ObjectTileCollision(CMODE_FLOOR,0,8,0)
		if CheckResult==1
			Object.State++
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
		endif
		Object.Value0=Object.YPos
		Object.Value0&=-0x10000
		Object.Value0-=TempValue0
		TempValue1=Object.YPos
		Object.YPos=TempValue0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].YPos+=Object.Value0
			endif
			if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
				if CheckResult==4
					CallFunction(PlayerObject_Kill)
				endif
			endif
		floop
		Object.YPos=TempValue1
		break
	case 4
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		break

	endswitch
endsub


sub ObjectDraw
	if Object.State>0
		DrawSprite(0)
	endif
endsub


sub ObjectStartup
	LoadSpriteSheet("LZ/Objects.gif")
	SpriteFrame(-16,-8,32,16,1,1)
	foreach TypeName[MovingBlock],ArrayPos0
		Object[ArrayPos0].Priority=PRIORITY_XBOUNDS
	floop
	SetArrayValue(TypeName[MovingBlock],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(MovingBlock_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(MovingBlock_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
