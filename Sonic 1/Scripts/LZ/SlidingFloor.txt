function SlidingFloor_Function170
	CreateTempObject(TypeName[SlidingFloor],0,Object.XPos,Object.YPos)
endfunction


sub ObjectMain
	switch Object.State
	case 0
		if Object[+1].Value0==1
			Object.State++
			Object.Value0=256
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,2,2,2,2)
			if CheckResult==1
				if Object[PlayerObjectPos].YPos>Object.YPos
					Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
					FlipSign(Object[PlayerObjectPos].XPos)
					Object[PlayerObjectPos].XPos-=0x800000
					Object[PlayerObjectPos].XPos+=Object.XPos
				else
					Object[PlayerObjectPos].iYPos=Object[PlayerObjectPos].CollisionBottom
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos-=0x400000
					Object[PlayerObjectPos].YPos+=Object.YPos
				endif
			endif
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-130,-63,130,63,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				if Object[PlayerObjectPos].XPos<Object.XPos
					if Object[PlayerObjectPos].Right==1
						Object[PlayerObjectPos].Pushing=2
						Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
						FlipSign(Object[PlayerObjectPos].XPos)
						Object[PlayerObjectPos].XPos-=0x800000
						Object[PlayerObjectPos].XPos+=Object.XPos
						if Object[PlayerObjectPos].YPos==0x16D0000
							Object[PlayerObjectPos].YPos-=0x10000
						endif
						if Object[PlayerObjectPos].YPos==0x1710000
							Object[PlayerObjectPos].YPos-=0x10000
						endif
					endif
				endif
			endif
		floop
		break
	case 2
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,2,2,2,2)
			if CheckResult==1
				if Object[PlayerObjectPos].YPos>Object.YPos
					Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
					FlipSign(Object[PlayerObjectPos].XPos)
					Object[PlayerObjectPos].XPos-=0x800000
					Object[PlayerObjectPos].XPos+=Object.XPos
				else
					Object[PlayerObjectPos].iYPos=Object[PlayerObjectPos].CollisionBottom
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos-=0x400000
					Object[PlayerObjectPos].YPos+=Object.YPos
				endif
			endif
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-130,-63,130,63,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				if Object[PlayerObjectPos].XPos<Object.XPos
					if Object[PlayerObjectPos].Right==1
						Object[PlayerObjectPos].Pushing=2
						Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
						FlipSign(Object[PlayerObjectPos].XPos)
						Object[PlayerObjectPos].XPos-=0x800000
						Object[PlayerObjectPos].XPos+=Object.XPos
						if Object[PlayerObjectPos].YPos==0x16D0000
							Object[PlayerObjectPos].YPos-=0x10000
						endif
						if Object[PlayerObjectPos].YPos==0x1710000
							Object[PlayerObjectPos].YPos-=0x10000
						endif
					endif
				endif
			endif
		floop
		break
	case 1
		TempValue0=Object.XPos
		TempValue0&=-0x10000
		TempValue1=Object.YPos
		TempValue1&=-0x10000
		Object.XPos-=0x10000
		Object.YPos+=0x8000
		Object.Value0--
		if Object.Value0<0
			Object.State++
		endif
		Object.Value1=Object.XPos
		Object.Value1&=-0x10000
		Object.Value1-=TempValue0
		Object.Value2=Object.YPos
		Object.Value2&=-0x10000
		Object.Value2-=TempValue1
		TempValue2=Object.XPos
		Object.XPos=TempValue0
		TempValue3=Object.YPos
		Object.YPos=TempValue1
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].XPos+=Object.Value1
				Object[PlayerObjectPos].YPos+=Object.Value2
			endif
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,2,2,2,2)
			if CheckResult==1
				if Object[PlayerObjectPos].YPos>Object.YPos
					Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
					FlipSign(Object[PlayerObjectPos].XPos)
					Object[PlayerObjectPos].XPos-=0x800000
					Object[PlayerObjectPos].XPos+=Object.XPos
				else
					Object[PlayerObjectPos].iYPos=Object[PlayerObjectPos].CollisionBottom
					FlipSign(Object[PlayerObjectPos].YPos)
					Object[PlayerObjectPos].YPos-=0x400000
					Object[PlayerObjectPos].YPos+=Object.YPos
				endif
			endif
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-130,-63,130,63,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				if Object[PlayerObjectPos].XPos<Object.XPos
					if Object[PlayerObjectPos].Right==1
						Object[PlayerObjectPos].Pushing=2
						Object[PlayerObjectPos].iXPos=Object[PlayerObjectPos].CollisionRight
						FlipSign(Object[PlayerObjectPos].XPos)
						Object[PlayerObjectPos].XPos-=0x800000
						Object[PlayerObjectPos].XPos+=Object.XPos
					endif
				endif
			endif
		floop
		Object.XPos=TempValue2
		Object.YPos=TempValue3
		break

	endswitch
endsub
