#alias 54: TYPE_FALLINGBLOCK

// Function declarations
#function FallingBlock_Function128
#function FallingBlock_Function129
#function FallingBlock_Function130

#constant 0:FallingBlock_constant33


function FallingBlock_Function128
	DrawSprite(0)
endfunction


function FallingBlock_Function129
	CreateTempObject(TypeName[FallingBlock],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].State=Object[0].Value17
	Object[TempObjectPos].State-=FallingBlock_constant33
	switch Object[TempObjectPos].State
	case 0
		Object[TempObjectPos].Value4=0x400
		break
	case 1
	case 2
		Object[TempObjectPos].YPos-=0x80000
		Object[TempObjectPos].Value4=0x400
		break
	case 3
		Object[TempObjectPos].State++
		Object[TempObjectPos].Value4=512
		break

	endswitch
endfunction


function FallingBlock_Function130
	Sin(Object.Value2,Object.Angle)
	Object.Value2*=Object.Value4
	Object.Value2+=Object.YPos
	Object.Value2&=-0x10000
	Object.Value2-=Object.Value1
	TempValue0=Object.YPos
	Object.YPos=Object.Value1
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		switch CheckResult
		case 1
			Object[PlayerObjectPos].YPos+=Object.Value2
			break
		case 4
			if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
				CallFunction(PlayerObject_Function51)
			endif
			break

		endswitch
	floop
	Object.YPos=TempValue0
	Sin(Object.Value1,Object.Angle)
	Object.Value1*=Object.Value4
	Object.Value1+=Object.YPos
	Object.Value1&=-0x10000
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Value1=Object.YPos
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		break
	case 1
		Object.Priority=PRIORITY_ACTIVE
		Object.Angle=oscillation
		Object.Angle<<=2
		CallFunction(FallingBlock_Function130)
		break
	case 2
		Object.Priority=PRIORITY_ACTIVE
		Object.Angle=oscillation
		Object.Angle<<=2
		CallFunction(FallingBlock_Function130)
		foreach TypeGroup[256],PlayerObjectPos
			if Object[PlayerObjectPos].YPos>Object.YPos
				TempValue0=Object.XPos
				TempValue0-=Object[PlayerObjectPos].XPos
				Absolute(TempValue0)
				if TempValue0<0x900000
					Object.State=3
					Object.YPos=Object.Value1
					Object.Value4=512
				endif
			endif
		floop
		break
	case 3
		Object.Priority=PRIORITY_ACTIVE
		Object.YVelocity+=0x1800
		Object.YPos+=Object.YVelocity
		Object.Value1=Object.YPos
		ObjectTileCollision(CMODE_FLOOR,0,16,0)
		if CheckResult==1
			TempValue1=Object.iYPos
			TempValue1+=8
			Get16x16TileInfo(TempValue0,Object.iXPos,TempValue1,TILEINFO_ANGLEB)
			if TempValue0==1
				Object.State=4
			else
				Object.State=0
			endif
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==4
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					CallFunction(PlayerObject_Function51)
				endif
			endif
		floop
		break
	case 4
		Object.Priority=PRIORITY_ACTIVE
		Object.Angle=oscillation
		Object.Angle<<=1
		CallFunction(FallingBlock_Function130)
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.YPos
		Object.YPos=Object.Value3
		if Object.OutOfBounds==1
			Object.YVelocity=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.State=Object.PropertyValue
			Object.State&=7
			if Object.State>4
				Object.State=0
			endif
			switch Object.State
			case 0
			case 1
			case 2
				Object.Value4=0x400
				break
			case 4
				Object.Value4=512
				break

			endswitch
		else
			Object.YPos=TempValue0
		endif
	endif
endsub


sub ObjectDraw
	DrawSpriteXY(0,Object.XPos,Object.Value1)
endsub


sub ObjectStartup
	LoadSpriteSheet("MZ/Objects.gif")
	SpriteFrame(-16,-16,32,32,191,289)
	foreach TypeName[FallingBlock],ArrayPos0
		Object[ArrayPos0].State=Object[ArrayPos0].PropertyValue
		Object[ArrayPos0].State&=7
		if Object[ArrayPos0].State>4
			Object[ArrayPos0].State=0
		endif
		switch Object[ArrayPos0].State
		case 0
			Object[ArrayPos0].Value4=0x400
			break
		case 1
		case 2
			Object[ArrayPos0].YPos-=0x80000
			Object[ArrayPos0].Value4=0x400
			break
		case 4
			Object[ArrayPos0].Value4=512
			break

		endswitch
		Object[ArrayPos0].Value3=Object[ArrayPos0].YPos
	floop
	TempValue0=0
	FallingBlock_constant33=DebugMode_DebugItemCount
	while TempValue0<4
		SetArrayValue(54,DebugMode_DebugItemCount,DebugMode_DebugTypeList)
		SetArrayValue(128,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
		SetArrayValue(129,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
		DebugMode_DebugItemCount++
		TempValue0++
	loop
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
