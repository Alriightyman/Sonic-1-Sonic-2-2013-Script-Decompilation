#alias 59: TYPE_RIDEBLOCK

// Function declarations
#function RideBlock_DebugDraw
#function RideBlock_DebugSpawn



function RideBlock_DebugDraw
	DrawSprite(0)
endfunction


function RideBlock_DebugSpawn
	CreateTempObject(TypeName[RideBlock],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Direction=Object.Direction
	Object[TempObjectPos].Value2=Object[ArrayPos0].XPos
	Object[TempObjectPos].Value3=Object[ArrayPos0].YPos
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-16,-16,16,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object.State++
				if Object.Direction==FLIP_NONE
					Object.XVelocity=0x10000
				else
					Object.XVelocity=-0x10000
				endif
			endif
		floop
		break
	case 1
		TempValue0=Object.XPos
		TempValue0-=Object.Value1
		TempValue0+=Object.XVelocity
		TempValue0&=-0x10000
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-16,-16,16,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object[PlayerObjectPos].XPos+=TempValue0
			endif
		floop
		Object.XPos+=Object.XVelocity
		if Object.XVelocity>0
			ObjectTileCollision(CMODE_LWALL,16,12,0)
			if CheckResult==1
				Object.XVelocity=0
				Object.Value0=0
				TempValue1=Object.iYPos
				TempValue1+=12
				Get16x16TileInfo(TempValue0,Object.iXPos,TempValue1,TILEINFO_ANGLEB)
				if TempValue0==1
					Object.State=2
				else
					Object.State=3
				endif
			endif
		else
			ObjectTileCollision(CMODE_RWALL,-16,12,0)
			if CheckResult==1
				Object.XVelocity=0
				Object.Value0=0
				TempValue1=Object.iYPos
				TempValue1+=12
				Get16x16TileInfo(TempValue0,Object.iXPos,TempValue1,TILEINFO_ANGLEB)
				if TempValue0==1
					Object.State=2
				else
					Object.State=3
				endif
			endif
		endif
		break
	case 2
		if Object.Value0<320
			Object.Value0++
			TempValue0=Object.Value0
			TempValue0&=7
			if TempValue0==0
				TempValue0=0x10000
			else
				TempValue0=0
			endif
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-16,-16,16,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if CheckResult==1
					Object[PlayerObjectPos].YPos+=TempValue0
				endif
			floop
			Object.YPos+=TempValue0
		endif
		break
	case 3
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-16,-16,16,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		break

	endswitch
	Object.Value1=Object.XPos
	Object.Value1&=-0x10000
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value2
		Object.YPos=Object.Value3
		if Object.OutOfBounds==1
			Object.YVelocity=0
			Object.State=0
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
endsub


sub ObjectDraw
	DrawSpriteXY(0,Object.Value1,Object.YPos)
endsub


sub ObjectStartup
	LoadSpriteSheet("MZ/Objects.gif")
	SpriteFrame(-16,-16,32,32,159,322)
	foreach TypeName[RideBlock],ArrayPos0
		Object[ArrayPos0].Direction=Object[ArrayPos0].PropertyValue
		Object[ArrayPos0].Direction&=FLIP_X
		Object[ArrayPos0].Value2=Object[ArrayPos0].XPos
		Object[ArrayPos0].Value3=Object[ArrayPos0].YPos
	floop
	SetArrayValue(TypeName[RideBlock],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(RideBlock_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(RideBlock_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
