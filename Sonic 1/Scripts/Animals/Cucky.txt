function Cucky_DebugDraw
	DrawSprite(2)
endfunction


function Cucky_DebugSpawn
	CreateTempObject(TypeName[Cucky],0,Object.XPos,Object.YPos)
	GetBit(TempValue0,Object.Direction,0)
	if TempValue0==0
		Object[TempObjectPos].State=3
		Object[TempObjectPos].XVelocity=0x20000
		Object[TempObjectPos].YVelocity=-0x30000
		Rand(CheckResult,32)
		if CheckResult<16
			Object[TempObjectPos].Direction=FLIP_X
			FlipSign(Object[TempObjectPos].XVelocity)
		endif
	else
		Object[TempObjectPos].State=4
	endif
endfunction


sub ObjectMain
	switch Object.State
	case 0
		if Object.Value0<Object.PropertyValue
			Object.Value0++
		else
			Object.XVelocity=0x20000
			if Object.Value1==1
				Object.DrawOrder=4
				Rand(CheckResult,32)
				if CheckResult<16
					Object.Direction=FLIP_X
					FlipSign(Object.XVelocity)
				endif
			else
				Object.Direction=FLIP_X
				FlipSign(Object.XVelocity)
			endif
			Object.YVelocity=-0x40000
			Object.State++
		endif
		break
	case 1
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,8,0)
			if CheckResult==1
				Object.YVelocity=-0x30000
				Object.State++
			endif
		endif
		break
	case 4
		if Object[0].XPos<Object.XPos
			Object.Direction=FLIP_X
		else
			Object.Direction=FLIP_NONE
		endif
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,8,0)
			if CheckResult==1
				if Object.State==3
					if Object.Value0==1
						FlipSign(Object.XVelocity)
						Object.Direction^=FLIP_X
					endif
					Object.Value0^=1
				endif
				Object.YVelocity=-0x30000
			endif
		endif
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=1
		Object.AnimationTimer++
		Object.AnimationTimer&=3
		break
	case 2
	case 3
	case 5
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.YVelocity>0
			ObjectTileCollision(CMODE_FLOOR,0,8,0)
			if CheckResult==1
				if Object.State==3
					if Object.Value0==1
						FlipSign(Object.XVelocity)
						Object.Direction^=FLIP_X
					endif
					Object.Value0^=1
				endif
				Object.YVelocity=-0x30000
			endif
		endif
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=1
		Object.AnimationTimer++
		Object.AnimationTimer&=3
		break

	endswitch
	if Object.State<3
		if Object.OutOfBounds==1
			Object.Type=TypeName[BlankObject]
		endif
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 0
	case 1
		DrawSprite(2)
		break
	case 2
	case 3
	case 4
	case 5
		DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
		break

	endswitch
endsub


sub ObjectStartup
	LoadSpriteSheet("Global/Items.gif")
	SpriteFrame(-8,-8,16,16,174,215)
	SpriteFrame(-8,-8,16,16,97,227)
	SpriteFrame(-7,-12,15,24,207,197)
	foreach TypeName[Cucky],ArrayPos0
		if Object[ArrayPos0].PropertyValue==1
			Object[ArrayPos0].State=3
			Object[ArrayPos0].XVelocity=0x20000
			Object[ArrayPos0].YVelocity=-0x30000
			Rand(CheckResult,32)
			if CheckResult<16
				Object[ArrayPos0].Direction=FLIP_X
				FlipSign(Object[ArrayPos0].XVelocity)
			endif
		else
			if Object[ArrayPos0].PropertyValue==2
				Object[ArrayPos0].State=4
			endif
			if Object[ArrayPos0].PropertyValue==3
				Object[ArrayPos0].State=5
				Object[ArrayPos0].XVelocity=0x20000
				FlipSign(Object[ArrayPos0].XVelocity)
				Object[ArrayPos0].Direction=FLIP_X
			endif
		endif
		Object[ArrayPos0].PropertyValue=0
	floop
	SetArrayValue(TypeName[Cucky],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(Cucky_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(Cucky_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub
