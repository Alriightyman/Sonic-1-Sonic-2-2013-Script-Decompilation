function VBlock_DebugDraw
	DrawSprite(0)
endfunction


function VBlock_DebugSpawn
	CreateTempObject(TypeName[VBlock],Object.Direction,Object.XPos,Object.YPos)
	if Object[TempObjectPos].PropertyValue==1
		Object[TempObjectPos].YPos-=0x200000
	else
		Object[TempObjectPos].YPos-=0x1F0000
	endif
endfunction


sub ObjectMain
	Object.Angle=SYZSetup_constant30
	Object.Angle*=512
	Object.Angle/=356
	if Object.PropertyValue==1
		Object.Angle+=256
	endif
	Object.Angle&=511
	Sin(Object.Value2,Object.Angle)
	Object.Value2<<=12
	Object.Value2+=Object.YPos
	Object.Value2&=-0x10000
	Object.Value2-=Object.Value1
	foreach TypeGroup[256],PlayerObjectPos
		TempValue0=Object.YPos
		Object.YPos=Object.Value1
		PlayerObjectCollision(C_BOX,Object.EntityNo,-32,-32,32,32,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		switch CheckResult
		case 1
			Object[PlayerObjectPos].YPos+=Object.Value2
			break
		case 4
			if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
				if Object.Angle>384
					CallFunction(PlayerObject_Function51)
				endif
				if Object.Angle<128
					CallFunction(PlayerObject_Function51)
				endif
			endif
			endswitch
			Object.YPos=TempValue0
		floop
endsub



sub ObjectDraw
	Sin(Object.Value1,Object.Angle)
	Object.Value1<<=12
	Object.Value1+=Object.YPos
	Object.Value1&=-0x10000
	DrawSpriteXY(0,Object.XPos,Object.Value1)
endsub


sub ObjectStartup
	LoadSpriteSheet("SYZ/Objects.gif")
	SpriteFrame(-32,-32,64,64,119,34)
	foreach TypeName[VBlock],ArrayPos0
		if Object[ArrayPos0].PropertyValue==1
			Object[ArrayPos0].YPos-=0x200000
		else
			Object[ArrayPos0].YPos-=0x1F0000
		endif
	floop
	SetArrayValue(TypeName[VBlock],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(VBlock_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(VBlock_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
