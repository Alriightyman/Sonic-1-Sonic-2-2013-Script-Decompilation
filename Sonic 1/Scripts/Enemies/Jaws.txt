function Jaws_Function115
	CreateTempObject(TypeName[Jaws],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Direction=Object.Direction
	Object[TempObjectPos].Value3=Object.XPos
	if Object[TempObjectPos].Direction==FLIP_NONE
		Object[TempObjectPos].XVelocity=-0x4000
	else
		Object[TempObjectPos].XVelocity=0x4000
	endif
	Object[TempObjectPos].Value0=120
endfunction


sub ObjectMain
	Object.Priority=PRIORITY_ACTIVE
	Object.XPos+=Object.XVelocity
	Object.Value0--
	if Object.Value0==0
		Object.Direction^=FLIP_X
		FlipSign(Object.XVelocity)
		Object.Value0=Object.Value1
	endif
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		Object.XPos=Object.Value3
		if Object.OutOfBounds==1
			Object.XPos=Object.Value3
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.Direction=Object.PropertyValue
			Object.Direction>>=7
			if Object.Direction==FLIP_NONE
				Object.XVelocity=-0x4000
			else
				Object.XVelocity=0x4000
			endif
			Object.Value0=Object.PropertyValue
			Object.Value0&=127
			Object.Value0<<=6
			Object.Value1=Object.Value0
		else
			Object.XPos=TempValue0
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-12,16,12,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function45)
		endif
	floop
	Object.Frame=Object.AnimationTimer
	Object.Frame>>=3
	Object.AnimationTimer++
	Object.AnimationTimer&=15
endsub
