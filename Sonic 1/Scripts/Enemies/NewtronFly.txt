function NewtronFly_Function130
	CreateTempObject(TypeName[NewtronFly],Object.Direction,Object.XPos,Object.YPos)
	Object[TempObjectPos].InkEffect=INK_ALPHA
	Object[TempObjectPos].Alpha=0
	Object[TempObjectPos].Value1=Object.XPos
	Object[TempObjectPos].Value2=Object.YPos
endfunction


sub ObjectMain
	switch Object.State
	case 0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-128,-64,128,64,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object.State=1
			endif
		floop
		break
	case 1
		Object.Priority=PRIORITY_ACTIVE
		if Object.Alpha<248
			Object.Alpha+=4
			if Object[0].XPos>Object.XPos
				Object.PropertyValue=0
			else
				Object.PropertyValue=1
			endif
		else
			Object.Alpha=255
			Object.Animation=1
			if Object.PropertyValue==0
				Object.XVelocity=0x20000
			else
				Object.XVelocity=-0x20000
			endif
			Object.Direction=Object.PropertyValue
			Object.State++
		endif
		break
	case 2
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		ObjectTileCollision(CMODE_FLOOR,0,8,0)
		if CheckResult==1
			Object.YVelocity=0
			Object.AnimationTimer=0
			Object.Animation=2
			Object.State++
		endif
		break
	case 3
		Object.XPos+=Object.XVelocity
		ObjectTileGrip(CMODE_FLOOR,0,8,0)
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value1
		Object.YPos=Object.Value2
		if Object.OutOfBounds==1
			Object.YVelocity=0
			Object.XPos=Object.Value1
			Object.YPos=Object.Value2
			Object.Animation=0
			Object.AnimationTimer=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.Alpha=0
			Object.State=0
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		switch Object.State
		case 2
		case 3
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-16,-8,16,8,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
			if CheckResult==1
				CallFunction(PlayerObject_Function45)
			endif
			break

		endswitch
	floop
	switch Object.Animation
	case 1
		if Object.AnimationTimer<41
			Object.Frame=Object.AnimationTimer
			Object.Frame/=20
			Object.AnimationTimer++
		endif
		Object.Frame+=2
		break
	case 2
		Object.Frame=Object.AnimationTimer
		Object.Frame/=3
		Object.Frame+=5
		Object.AnimationTimer++
		Object.AnimationTimer%=6
		break

	endswitch
endsub


sub ObjectDraw
	switch Object.Animation
	case 0
		DrawSpriteFX(Object.PropertyValue,FX_INK,Object.XPos,Object.YPos)
		break
	case 1
		DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 2
		DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
		DrawSpriteFX(4,FX_FLIP,Object.XPos,Object.YPos)
		break

	endswitch
endsub
