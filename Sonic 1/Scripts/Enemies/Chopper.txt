function Chopper_Function123
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Chopper_Function124
	CreateTempObject(TypeName[Chopper],0,Object.XPos,Object.YPos)
	GetBit(TempValue0,Object.Direction,0)
	Object[TempObjectPos].Direction=TempValue0
	Object[TempObjectPos].Value1=Object.YPos
	Object[TempObjectPos].YVelocity=-0x70000
endfunction


sub ObjectMain
	Object.YPos+=Object.YVelocity
	Object.YVelocity+=0x1800
	if Object.YVelocity<-0x38000
		Object.Animation=0
	else
		if Object.YVelocity>0x38000
			Object.Animation=2
		else
			Object.Animation=1
		endif
	endif
	if Object.YPos>Object.Value1
		Object.YPos=Object.Value1
		Object.YVelocity=-0x70000
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-12,-14,12,14,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_Function45)
		endif
	floop
	switch Object.Animation
	case 0
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=3
		Object.AnimationTimer++
		Object.AnimationTimer&=15
		break
	case 1
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=3
		Object.AnimationTimer+=2
		Object.AnimationTimer&=15
		break
	case 2
		Object.Frame=0
		break

	endswitch
endsub


sub ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
endsub

	CheckStageFolder("Zone01")
