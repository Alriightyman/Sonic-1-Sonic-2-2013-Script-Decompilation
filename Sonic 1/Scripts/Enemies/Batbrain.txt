#alias 49: TYPE_BATBRAIN

// Function declarations
#function Batbrain_DebugDraw
#function Batbrain_DebugSpawn



function Batbrain_DebugDraw
	DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
endfunction


function Batbrain_DebugSpawn
	CreateTempObject(TypeName[Batbrain],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value0=Object.XPos
	Object[TempObjectPos].Value1=Object.YPos
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		Object.Value4=0x7FFFFFFF
		Object.Value2=0
		Object.Value3=0xFFFF
		foreach TypeGroup[256],PlayerObjectPos
			TempValue0=Object[PlayerObjectPos].XPos
			TempValue0-=Object.XPos
			Absolute(TempValue0)
			if TempValue0<Object.Value4
				Object.Value4=TempValue0
				Object.Value2=PlayerObjectPos
			endif
			if TempValue0<0x800000
				if Object[PlayerObjectPos].YPos>=Object.YPos
					if Object.Value3!=0xFFFF
						ArrayPos0=Object.Value3
						if Object[PlayerObjectPos].YPos<Object[ArrayPos0].YPos
							Object.Value3=PlayerObjectPos
						endif
					else
						Object.Value3=PlayerObjectPos
					endif
				endif
			endif
		floop
		if Object.Value3!=0xFFFF
			ArrayPos0=Object.Value3
			TempValue0=Object[ArrayPos0].YPos
			TempValue0-=Object.YPos
			if TempValue0>=0
				if TempValue0<=0x800000
					Rand(TempValue0,8)
					if TempValue0==0
						Object.State=1
						Object.Value5=Object[ArrayPos0].YPos
						Object.Animation=1
					endif
				endif
			endif
		else
			ArrayPos0=Object.Value2
		endif
		if Object[ArrayPos0].XPos<Object.XPos
			Object.Direction=FLIP_NONE
		else
			Object.Direction=FLIP_X
		endif
		break
	case 1
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x1800
		ArrayPos0=Object.Value3
		if Object[ArrayPos0].XPos<Object.XPos
			Object.Direction=FLIP_NONE
		else
			Object.Direction=FLIP_X
		endif
		TempValue0=Object.Value5
		TempValue0-=Object.YPos
		if TempValue0<0x100000
			if Object.Direction==FLIP_NONE
				Object.XVelocity=-0x10000
			else
				Object.XVelocity=0x10000
			endif
			Object.YVelocity=0
			Object.Animation=2
			Object.State=2
		endif
		break
	case 2
		Object.XPos+=Object.XVelocity
		ArrayPos0=Object.Value3
		TempValue0=Object[ArrayPos0].XPos
		TempValue0-=Object.XPos
		Absolute(TempValue0)
		if TempValue0>=0x800000
			Rand(TempValue0,8)
			if TempValue0==0
				Object.State=3
			endif
		endif
		break
	case 3
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		Object.YVelocity-=0x1800
		ObjectTileCollision(CMODE_ROOF,0,-12,0)
		if CheckResult==1
			Object.XVelocity=0
			Object.YVelocity=0
			Object.State=0
			Object.Animation=0
		endif
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value0
		Object.YPos=Object.Value1
		if Object.OutOfBounds==1
			Object.XVelocity=0
			Object.YVelocity=0
			Object.Direction=FLIP_NONE
			Object.Animation=0
			Object.Frame=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.State=0
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-12,-18,12,18,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_BadnikBreak)
		endif
	floop
	switch Object.Animation
	case 0
		Object.Frame=0
		break
	case 1
		Object.Frame=1
		break
	case 2
		if Object.AnimationTimer==0
			PlaySfx(SFXName[BatbrainFly],0)
		endif
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=2
		Object.Frame++
		Object.AnimationTimer++
		Object.AnimationTimer&=15
		break

	endswitch
endsub


sub ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
endsub


sub ObjectStartup
	CheckStageFolder("Zone02")
	if CheckResult==1
		LoadSpriteSheet("MZ/Objects.gif")
		SpriteFrame(-7,-12,14,24,37,98)
		SpriteFrame(-12,-14,32,30,52,81)
		SpriteFrame(-15,-6,35,20,1,106)
		SpriteFrame(-12,-8,29,28,85,81)
		SpriteFrame(-15,-6,35,20,1,106)
	endif
	CheckStageFolder("Zone07")
	if CheckResult==1
		LoadSpriteSheet("MBZ/Objects.gif")
		SpriteFrame(-7,-12,14,24,52,68)
		SpriteFrame(-12,-14,32,30,1,93)
		SpriteFrame(-15,-6,35,20,64,99)
		SpriteFrame(-12,-8,29,28,34,93)
		SpriteFrame(-15,-6,35,20,64,99)
	endif
	foreach TypeName[Batbrain],ArrayPos0
		Object[ArrayPos0].Value0=Object[ArrayPos0].XPos
		Object[ArrayPos0].Value1=Object[ArrayPos0].YPos
		Object[ArrayPos0].DrawOrder=5
	floop
	SetArrayValue(TypeName[Batbrain],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(Batbrain_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(Batbrain_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
