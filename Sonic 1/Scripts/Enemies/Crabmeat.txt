#function Crabmeat_DebugDraw
#function Crabmeat_DebugSpawn



function Crabmeat_DebugDraw
	DrawSprite(0)
endfunction


function Crabmeat_DebugSpawn
	CreateTempObject(TypeName[Crabmeat],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value1=Object.XPos
	Object[TempObjectPos].Value2=Object.YPos
	GetBit(TempValue0,Object.Direction,0)
	if TempValue0==0
		Object[TempObjectPos].Direction=FLIP_X
		Object[TempObjectPos].XVelocity=0x8000
	else
		Object[TempObjectPos].Direction=FLIP_X
		Object[TempObjectPos].XVelocity=-0x8000
	endif
	Object[TempObjectPos].Value3=0
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Priority=PRIORITY_ACTIVE
		if Object.Value0<127
			Object.Value0++
			Object.XPos+=Object.XVelocity
			if Object.XVelocity>0
				ObjectTileGrip(CMODE_FLOOR,14,15,0)
			else
				ObjectTileGrip(CMODE_FLOOR,-14,15,0)
			endif
			if CheckResult==0
				if Object.Value3==1
					Object.State=2
				else
					Object.State=1
					Object.Value3=1
				endif
				Object.Value0=0
				Object.Animation=1
			endif
			ObjectTileGrip(CMODE_FLOOR,0,15,0)
		else
			if Object.Value3==1
				Object.State=2
			else
				Object.State=1
				Object.Value3=1
			endif
			Object.Value0=0
			Object.Animation=1
		endif
		break
	case 1
		if Object.Value0<59
			Object.Value0++
		else
			Object.Value0=0
			Object.State=0
			Object.Animation=0
			Object.AnimationTimer=0
			FlipSign(Object.XVelocity)
		endif
		break
	case 2
		if Object.Value0<59
			Object.Value0++
		else
			CreateTempObject(TypeName[CrabmeatShot],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos-=0x100000
			Object[TempObjectPos].XVelocity=-0x10000
			Object[TempObjectPos].YVelocity=-0x40000
			CreateTempObject(TypeName[CrabmeatShot],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].XPos+=0x100000
			Object[TempObjectPos].XVelocity=0x10000
			Object[TempObjectPos].YVelocity=-0x40000
			Object.Value0=0
			Object.State++
			Object.Animation=2
		endif
		break
	case 3
		if Object.Value0<59
			Object.Value0++
		else
			Object.Value0=0
			Object.State=0
			Object.Animation=0
			Object.AnimationTimer=0
			FlipSign(Object.XVelocity)
		endif
		break

	endswitch
	if Object.OutOfBounds==1
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		Object.XPos=Object.Value1
		Object.YPos=Object.Value2
		if Object.OutOfBounds==1
			if Object.PropertyValue==1
				Object.Direction=FLIP_X
				Object.XVelocity=0x8000
			else
				Object.Direction=FLIP_X
				Object.XVelocity=-0x8000
			endif
			Object.Value3=0
			Object.State=0
			Object.Value0=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
			Object.Animation=0
			Object.AnimationTimer=0
		else
			Object.XPos=TempValue0
			Object.YPos=TempValue1
		endif
	endif
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_TOUCH,Object.EntityNo,-14,-14,14,14,PlayerObjectPos,Object[PlayerObjectPos].Value40,Object[PlayerObjectPos].Value38,Object[PlayerObjectPos].Value41,Object[PlayerObjectPos].Value39)
		if CheckResult==1
			CallFunction(PlayerObject_BadnikBreak)
		endif
	floop
	switch Object.Animation
	case 0
		Object.Frame=Object.AnimationTimer
		Object.Frame>>=4
		Object.AnimationTimer++
		Object.AnimationTimer%=48
		break
	case 1
		Object.Frame=0
		break
	case 2
		Object.Frame=3
		break

	endswitch
endsub


sub ObjectDraw
	DrawSprite(Object.Frame)
endsub
