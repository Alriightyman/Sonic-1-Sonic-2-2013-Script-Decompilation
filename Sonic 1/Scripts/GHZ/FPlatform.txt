#function FallPlatform_Function103
#function FallPlatform_Function104



function FallPlatform_Function103
	DrawSprite(0)
endfunction


function FallPlatform_Function104
	CreateTempObject(TypeName[FallPlatform],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value3=Object[TempObjectPos].YPos
endfunction


sub ObjectMain
	if Object.Value0==1
		if Object.Value2==0
			Object.Priority=PRIORITY_ACTIVE
			Object.Value2=30
		endif
		if Object.Value4<64
			Object.Value4+=4
		endif
	else
		if Object.Value4>0
			Object.Value4-=4
		endif
	endif
	TempValue0=Object.YPos
	TempValue0&=-0x10000
	switch Object.State
	case 0
		TempValue1=Object.Value4
		TempValue1<<=1
		Sin(TempValue2,TempValue1)
		TempValue2<<=9
		Object.YPos=Object.Value3
		Object.YPos+=TempValue2
		if Object.PropertyValue==0
			if Object.Value2!=0
				Object.Value2--
				if Object.Value2==0
					Object.State++
					Object.Value2=32
				endif
			endif
		endif
		break
	case 1
		Object.Value2--
		if Object.Value2==0
			Object.State++
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-32,-8,32,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if CheckResult==1
					Object[PlayerObjectPos].YPos+=Object.YVelocity
					Object[PlayerObjectPos].YVelocity=Object.YVelocity
					Object[PlayerObjectPos].YVelocity-=0x3800
					Object[PlayerObjectPos].Gravity=GRAVITY_AIR
				endif
			floop
		endif
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.OutOfBounds==1
			Object.YPos=Object.Value3
			Object.YVelocity=0
			Object.Value2=0
			Object.State++
		endif
		break
	case 2
		Object.YPos+=Object.YVelocity
		Object.YVelocity+=0x3800
		if Object.OutOfBounds==1
			Object.YPos=Object.Value3
			Object.YVelocity=0
			Object.Value2=0
			Object.State++
		endif
		break
	case 3
		if Object.OutOfBounds==1
			Object.State=0
			Object.Priority=PRIORITY_ACTIVE_BOUNDS
		endif
		break

	endswitch
	Object.Value1=Object.YPos
	Object.Value1&=-0x10000
	Object.Value1-=TempValue0
	Object.Value0=0
	if Object.State<2
		TempValue1=Object.YPos
		Object.YPos=TempValue0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-32,-8,32,0,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object.Value0=1
				Object[PlayerObjectPos].YPos+=Object.Value1
			endif
		floop
		Object.YPos=TempValue1
	endif
endsub


sub ObjectDraw
	if Object.State<3
		DrawSprite(0)
	endif
endsub


sub ObjectStartup
	LoadSpriteSheet("GHZ/Objects.gif")
	SpriteFrame(-32,-14,64,32,50,18)
	foreach TypeName[FallPlatform],ArrayPos0
		Object[ArrayPos0].Value3=Object[ArrayPos0].YPos
	floop
	SetArrayValue(47,DebugMode_constant23,DebugMode_array12)
	SetArrayValue(103,DebugMode_constant23,DebugMode_array13)
	SetArrayValue(104,DebugMode_constant23,DebugMode_array11)
	DebugMode_constant23++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
