#function VPlatform2_DebugDraw
#function VPlatform2_DebugSpawn



function VPlatform2_DebugDraw
	DrawSprite(0)
endfunction


function VPlatform2_DebugSpawn
	CreateTempObject(TypeName[VPlatform2],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value0=Object.YPos
	Object[TempObjectPos].PropertyValue=Object.Direction
endfunction


sub ObjectMain
	TempValue7=Object.YPos
	GetArrayValue(TempValue0,12,StageSetup_array9)
	if Object.PropertyValue==1
		FlipSign(TempValue0)
		TempValue0+=0x4000
	else
		TempValue0-=0x4000
	endif
	TempValue0<<=7
	Object.YPos=Object.Value0
	Object.YPos-=TempValue0
	TempValue1=Object.YPos
	TempValue7&=-0x10000
	Object.Value1=Object.YPos
	Object.Value1&=-0x10000
	Object.Value1-=TempValue7
	Object.YPos=TempValue7
	foreach TypeGroup[256],PlayerObjectPos
		PlayerObjectCollision(C_PLATFORM,Object.EntityNo,-32,-2,32,12,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		if CheckResult==1
			Object[PlayerObjectPos].YPos+=Object.Value1
		endif
	floop
	Object.YPos=TempValue1
endsub


sub ObjectDraw
	DrawSprite(0)
endsub


sub ObjectStartup
	LoadSpriteSheet("GHZ/Objects.gif")
	SpriteFrame(-32,-8,64,144,191,1)
	foreach TypeName[VPlatform2],ArrayPos0
		Object[ArrayPos0].Value0=Object[ArrayPos0].YPos
	floop
	SetArrayValue(TypeName[VPlatform2],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(VPlatform2_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(VPlatform2_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
