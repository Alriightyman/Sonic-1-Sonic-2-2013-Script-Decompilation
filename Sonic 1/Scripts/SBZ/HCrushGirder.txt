#alias 52: TYPE_HCRUSHGIRDER

// Function declarations
#function HCrushGirder_DebugDraw
#function HCrushGirder_DebugSpawn



function HCrushGirder_DebugDraw
	DrawSprite(0)
endfunction


function HCrushGirder_DebugSpawn
	CreateTempObject(TypeName[HCrushGirder],0,Object.XPos,Object.YPos)
	Object[TempObjectPos].Value1=Object.XPos
	Object[TempObjectPos].Value2=Object.YPos
endfunction


sub ObjectMain
	switch Object.State
	case 0
		Object.Value0++
		if Object.Value0==96
			Object.Value0=6
			Object.Value3=1
			Object.State=4
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-96,-24,96,24,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			switch CheckResult
			case 1
				Object[PlayerObjectPos].XPos+=0x10000
				break
			case 2
				Object[PlayerObjectPos].Pushing=0
				break
			endswitch
		floop
		Object.XPos+=0x10000
		break
	case 1
		Object.Value0++
		if Object.Value0==48
			Object.Value0=6
			Object.Value3=2
			Object.State=4
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-96,-24,96,24,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			switch CheckResult
			case 1
				Object[PlayerObjectPos].YPos+=0x10000
				break
			case 4
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					CallFunction(PlayerObject_Function51)
				endif
				break
			endswitch
		floop
		Object.YPos+=0x10000
		break
	case 2
		Object.Value0++
		if Object.Value0==96
			Object.Value0=6
			Object.Value3=3
			Object.State=4
		endif
		Object.YPos-=0x4000
		Object.XPos-=0x10000
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-96,-24,96,24,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			switch CheckResult
			case 1
				Object[PlayerObjectPos].XPos-=0x10000
				break
			case 4
				break
			endswitch
		floop
		break
	case 3
		Object.Value0++
		if Object.Value0==24
			Object.Value0=6
			Object.Value3=0
			Object.State=4
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-96,-24,96,24,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			switch CheckResult
			case 1
				Object[PlayerObjectPos].YPos-=0x10000
				break
			case 4
				break
			endswitch
		floop
		Object.YPos-=0x10000
		break
	case 4
		if Object.Value0>0
			Object.Value0--
		else
			Object.State=Object.Value3
		endif
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-96,-24,96,24,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==4
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					CallFunction(PlayerObject_Function51)
				endif
			endif
		floop
	endswitch
endsub



sub ObjectDraw
	DrawSprite(0)
endsub


sub ObjectStartup
	LoadSpriteSheet("SBZ/Objects.gif")
	SpriteFrame(-96,-24,192,48,319,1)
	foreach TypeName[HCrushGirder],ArrayPos0
		Object[ArrayPos0].Value1=Object[ArrayPos0].XPos
		Object[ArrayPos0].Value2=Object[ArrayPos0].YPos
	floop
	SetArrayValue(TypeName[HCrushGirder],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(HCrushGirder_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(HCrushGirder_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
