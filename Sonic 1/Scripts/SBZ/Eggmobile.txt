sub ObjectMain
	switch Object.State
	case 0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-24,-96,24,96,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		break
	case 1
		Object.Priority=PRIORITY_ACTIVE
		Object.Animation=2
		Object.Value0=1
		Object.YVelocity=-0x18000
		Object.State++
		break
	case 2
		Object.YPos+=Object.YVelocity
		Object.Value4++
		if Object.Value4>=52
			Object.Value2=1
			Object.XVelocity=0x18000
			Object.YVelocity=-0x1800
			Object.State++
		endif
		break
	case 4
		Object.Value5++
		TempValue0=Object.Value5
		TempValue0&=7
		if TempValue0==4
			Rand(TempValue0,96)
			TempValue0-=48
			TempValue0<<=16
			TempValue0+=Object.XPos
			Rand(TempValue1,48)
			TempValue1-=24
			TempValue1<<=16
			TempValue1+=Object.YPos
			CreateTempObject(TypeName[Explosion],0,TempValue0,TempValue1)
			Object[TempObjectPos].DrawOrder=4
			if Object.Value7<256
				PlaySfx(SFXName[Explosion],0)
			endif
		endif
		if Object.Value7<256
			Object.Value4++
			if Object.Value4>=30
				PlaySfx(SFXName[BossHit],0)
				Object.YVelocity=0x6000
				Object.Value4=0
			endif
		endif
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		TempValue0=Stage.NewXBoundary2
		TempValue0+=212
		TempValue0<<=16
		if Object.XPos>=TempValue0
			if options.gameMode==2
				Object.Type=TypeName[BlankObject]
				ResetObjectEntity(30,TypeName[ActFinish],0,0,0)
				Object[30].DrawOrder=6
				Object[30].Priority=PRIORITY_ACTIVE
				Stage.TimeEnabled=0
				PlayMusic(1)
			else
				Object.Value7+=4
				Music.Volume-=2
				SetScreenFade(0,0,0,Object.Value7)
				if Object.Value7==384
					if Stage.DebugMode==0
						if options.stageSelectFlag==0
							EngineCallback(SetLeaderboard,19,player.score)
						endif
					endif
					if options.gameMode==1
						ArrayPos1=options.saveSlot
						ArrayPos1<<=3
						ArrayPos1+=4
						SaveRAM[ArrayPos1]=20
						WriteSaveRAM()
					endif
					Stage.ActiveList=PRESENTATION_STAGE
					Stage.ListPos=1
					LoadStage()
				endif
			endif
		endif
		foreach TypeGroup[256],PlayerObjectPos
			if Object[PlayerObjectPos].XPos>=Object.Value6
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleGround)
				TempValue0=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleAir)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleRoll)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_RollingJump)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_Blank)
				TempValue0|=CheckResult
				if TempValue0==1
					Object[PlayerObjectPos].State=PlayerObject_Blank
					Object[PlayerObjectPos].Direction=FLIP_NONE
					Object[PlayerObjectPos].ControlMode=-1
					if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
						Object[PlayerObjectPos].XVelocity=0
						Object[PlayerObjectPos].Speed=0
						if Object.State==3
							Object[PlayerObjectPos].Animation=ANI_LOOKINGUP
						else
							Object[PlayerObjectPos].Animation=ANI_STOPPED
						endif
					else
						Object[PlayerObjectPos].Speed=Object[PlayerObjectPos].XVelocity
						Object[PlayerObjectPos].YVelocity+=0x3800
					endif
				else
					if Object[PlayerObjectPos].State==PlayerObject_KnuxGlideSlide
						Object[PlayerObjectPos].Speed=0
					endif
				endif
			endif
		floop
		break
	case 3
		Object.XPos+=Object.XVelocity
		Object.YPos+=Object.YVelocity
		TempValue0=Stage.NewXBoundary2
		TempValue0+=212
		TempValue0<<=16
		if Object.XPos>=TempValue0
			if options.gameMode==2
				Object.Type=TypeName[BlankObject]
				ResetObjectEntity(30,TypeName[ActFinish],0,0,0)
				Object[30].DrawOrder=6
				Object[30].Priority=PRIORITY_ACTIVE
				Stage.TimeEnabled=0
				PlayMusic(1)
			else
				Object.Value7+=4
				Music.Volume-=2
				SetScreenFade(0,0,0,Object.Value7)
				if Object.Value7==384
					if Stage.DebugMode==0
						if options.stageSelectFlag==0
							EngineCallback(SetLeaderboard,19,player.score)
						endif
					endif
					if options.gameMode==1
						ArrayPos1=options.saveSlot
						ArrayPos1<<=3
						ArrayPos1+=4
						SaveRAM[ArrayPos1]=20
						WriteSaveRAM()
					endif
					Stage.ActiveList=PRESENTATION_STAGE
					Stage.ListPos=1
					LoadStage()
				endif
			endif
		endif
		foreach TypeGroup[256],PlayerObjectPos
			if Object[PlayerObjectPos].XPos>=Object.Value6
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleGround)
				TempValue0=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleAir)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_HandleRoll)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_RollingJump)
				TempValue0|=CheckResult
				CheckEqual(Object[PlayerObjectPos].State,PlayerObject_Blank)
				TempValue0|=CheckResult
				if TempValue0==1
					Object[PlayerObjectPos].State=PlayerObject_Blank
					Object[PlayerObjectPos].Direction=FLIP_NONE
					Object[PlayerObjectPos].ControlMode=-1
					if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
						Object[PlayerObjectPos].XVelocity=0
						Object[PlayerObjectPos].Speed=0
						if Object.State==3
							Object[PlayerObjectPos].Animation=ANI_LOOKINGUP
						else
							Object[PlayerObjectPos].Animation=ANI_STOPPED
						endif
					else
						Object[PlayerObjectPos].Speed=Object[PlayerObjectPos].XVelocity
						Object[PlayerObjectPos].YVelocity+=0x3800
					endif
				else
					if Object[PlayerObjectPos].State==PlayerObject_KnuxGlideSlide
						Object[PlayerObjectPos].Speed=0
					endif
				endif
			endif
		floop
		break
	case 5
		Object.XPos=Object[+10].XPos
		Object.YPos=Object[+10].YPos
		if Object[+1].Type==TypeName[BlankObject]
			Object.Type=TypeName[BlankObject]
		endif
		break

	endswitch
	if Object.State==3
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_TOUCH,Object.EntityNo,-20,-16,20,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				CallFunction(PlayerObject_CheckHit)
				if CheckResult==1
					Object.State=4
					Object.Animation=3
					PlaySfx(SFXName[BossHit],0)
					Object.Value4=0
					if Stage.DebugMode==0
						if PlayerObjectPos==0
							EngineCallback(SetAchievement,10,100)
						endif
					endif
				endif
			endif
		floop
	endif
endsub


sub ObjectDraw
	switch Object.Animation
	case 0
		DrawSpriteFX(16,FX_FLIP,Object.XPos,Object.YPos)
		DrawSpriteFX(17,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 1
		DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 2
		TempValue0=Object.AnimationTimer
		TempValue0>>=2
		TempValue0+=1
		Object.AnimationTimer++
		Object.AnimationTimer&=7
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 3
		TempValue0=Object.AnimationTimer
		TempValue0>>=2
		TempValue0+=3
		Object.AnimationTimer++
		Object.AnimationTimer&=7
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		break

	endswitch
	switch Object.Value0
	case 0
		DrawSpriteFX(5,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 1
		Object.Value1++
		if Object.Value1>=60
			Object.Value0=2
		else
			TempValue0=Object.Value1
			TempValue0/=20
			TempValue0+=5
			DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		endif
		break
	case 2
		break

	endswitch
	if Object.Value2!=0
		TempValue0=Object.Value3
		TempValue0/=3
		TempValue0+=8
		Object.Value3++
		if Object.Value3>21
			Object.Value3=18
		endif
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
	endif
endsub
