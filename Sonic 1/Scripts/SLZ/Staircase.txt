#alias 51: TYPE_STAIRCASE

// Function declarations
#function Staircase_DebugDraw
#function Staircase_DebugSpawn



function Staircase_DebugDraw
	DrawSprite(0)
endfunction


function Staircase_DebugSpawn
	CreateTempObject(TypeName[Staircase],0,Object.XPos,Object.YPos)
	GetBit(TempValue0,Object.Direction,0)
	if TempValue0==0
		Object[TempObjectPos].Direction=FLIP_NONE
	else
		Object[TempObjectPos].Direction=FLIP_X
	endif
endfunction


sub ObjectMain
	Object.Priority=PRIORITY_ACTIVE
	switch Object.State
	case 0
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,112,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==1
				Object.Value0=32
				Object.State++
			endif
		floop
		break
	case 1
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,112,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		Object.Value0--
		if Object.Value0<0
			Object.Value0=128
			Object.State++
		endif
		break
	case 2
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		if Object.Direction==FLIP_NONE
			Object.XPos+=0x600000
			TempValue4=-0x200000
		else
			TempValue4=0x200000
		endif
		TempValue5=Object.Value1
		TempValue2=1
		while TempValue2<5
			Object.YPos+=TempValue5
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if CheckResult==1
					TempValue6=Object.YPos
					TempValue6&=-0x10000
					TempValue7=0x4000
					TempValue7*=TempValue2
					TempValue7+=Object.YPos
					TempValue7&=-0x10000
					TempValue7-=TempValue6
					Object[PlayerObjectPos].YPos+=TempValue7
				endif
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					if CheckResult==4
						CallFunction(PlayerObject_Kill)
					endif
				endif
			floop
			Object.XPos+=TempValue4
			TempValue2++
		loop
		Object.Value1+=0x4000
		Object.XPos=TempValue0
		Object.YPos=TempValue1
		Object.Value0--
		if Object.Value0<=0
			Object.State++
		endif
		break
	case 3
		TempValue0=Object.XPos
		TempValue1=Object.YPos
		if Object.Direction==FLIP_NONE
			Object.XPos+=0x600000
			TempValue4=-0x200000
		else
			TempValue4=0x200000
		endif
		TempValue5=Object.Value1
		TempValue2=1
		while TempValue2<5
			Object.YPos+=TempValue5
			foreach TypeGroup[256],PlayerObjectPos
				PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,16,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
				if Object[PlayerObjectPos].Gravity==GRAVITY_GROUND
					if CheckResult==4
						CallFunction(PlayerObject_Kill)
					endif
				endif
			floop
			Object.XPos+=TempValue4
			TempValue2++
		loop
		Object.XPos=TempValue0
		Object.YPos=TempValue1
		break
	case 4
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,112,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
			if CheckResult==4
				Object.Value0=59
				Object.State++
			endif
		floop
		break
	case 5
		foreach TypeGroup[256],PlayerObjectPos
			PlayerObjectCollision(C_BOX,Object.EntityNo,-16,-16,112,16,PlayerObjectPos,0x10000,0x10000,0x10000,0x10000)
		floop
		Object.Value0--
		if Object.Value0<0
			Object.Value0=128
			Object.State=2
		endif
		break

	endswitch
	if Object.OutOfBounds==1
		Object.Priority=PRIORITY_ACTIVE_BOUNDS
		Object.State=0
		Object.Value1=0
		if Object.PropertyValue==2
			Object.State=4
		endif
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 0
	case 1
	case 4
		DrawSprite(0)
		break
	case 2
	case 3
		TempValue2=Object.XPos
		TempValue3=Object.YPos
		if Object.Direction==FLIP_NONE
			TempValue2+=0x600000
			TempValue4=-0x200000
		else
			TempValue4=0x200000
		endif
		TempValue5=Object.Value1
		TempValue3+=TempValue5
		DrawSpriteXY(1,TempValue2,TempValue3)
		TempValue2+=TempValue4
		TempValue3+=TempValue5
		DrawSpriteXY(1,TempValue2,TempValue3)
		TempValue2+=TempValue4
		TempValue3+=TempValue5
		DrawSpriteXY(1,TempValue2,TempValue3)
		TempValue2+=TempValue4
		TempValue3+=TempValue5
		DrawSpriteXY(1,TempValue2,TempValue3)
		break
	case 5
		TempValue2=Object.XPos
		if Object.Direction==FLIP_NONE
			TempValue2+=0x600000
			TempValue4=-0x200000
		else
			TempValue4=0x200000
		endif
		GetBit(TempValue5,Object.Value0,2)
		TempValue5<<=16
		TempValue6=0
		while TempValue6<4
			TempValue3=Object.YPos
			TempValue3+=TempValue5
			TempValue5^=0x10000
			DrawSpriteXY(1,TempValue2,TempValue3)
			TempValue2+=TempValue4
			TempValue6++
		loop
		break

	endswitch
endsub


sub ObjectStartup
	LoadSpriteSheet("SLZ/Objects.gif")
	SpriteFrame(-16,-16,128,32,67,26)
	SpriteFrame(-16,-16,32,32,67,26)
	foreach TypeName[Staircase],ArrayPos0
		if Object[ArrayPos0].PropertyValue==2
			Object[ArrayPos0].State=4
		endif
	floop
	SetArrayValue(TypeName[Staircase],DebugMode_DebugItemCount,DebugMode_DebugTypeList)
	SetArrayValue(Staircase_DebugDraw,DebugMode_DebugItemCount,DebugMode_DebugDrawList)
	SetArrayValue(Staircase_DebugSpawn,DebugMode_DebugItemCount,DebugMode_DebugSpawnList)
	DebugMode_DebugItemCount++
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
